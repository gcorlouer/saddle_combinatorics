\section{Related works}

\section{Setup and main theorem}
The central result of this paper is that the algebraic variety of critical points of the loss landscape of a deep linear network admits a stratification parametrized by a finite set of geometric invariants which are given by some rank patterns. These rank patterns are the rank of a product made of weight matrices and some projection of the input-output covariance matrix. 
\begin{definition}[Deep Linear Network]
A $L$-layers Deep Linear Network (DLN) is a function:
\begin{align*}
    f: \ & \mathbb{R}^{d_0} \times \prod_{l=0}^{L-1}\mathbb{R}^{d_l \times d_{l+1}}  \to \mathbb{R}^{d_L} \\
       & (x,\theta = (W_1,...,W_L)) \mapsto \mu(\theta)x = W_L...W_1 x
\end{align*}
The dimension $d_0$ is the dimension of the input data and the dimensions $d_l,\ 1\leq l \leq L$ are the width of the layers. A Deep Linear Network is a multi-linear function (in parameter space $\Omega=\prod_{l=0}^{L-1}\mathbb{R}^{d_l \times d_{l+1}}$) which is linear in each of the weight matrix $W_l$ and the data. It is linear in function space $\mathcal{F}:=\operatorname{Mat}_{d_L\times d_0}$ over $\mu(\theta):=W:=W_L...W_1$ where $\mu$ is the product map. For the input-output data $(X,Y)\sim P$ we have the input and input-out covariance matrices $\Sigma_X = \mathbb{E}[XX^{\top}]$ and $\Sigma_{YX}=\mathbb{E}[YX^{\top}]$ and we will work with the quadratic population loss:
\[
L(\theta) := ||M - \mu(\theta)\Sigma_x^{-\frac{1}{2}}||^2_F
\]
where $||\cdot||_F$ is the Frobenius norm and $M:=\Sigma_{YX}\Sigma_X^{-1}$ is the teacher matrix which is the ordinary least square regressor. 
\end{definition}
We define some algebraic varieties that will be important to derive the main result. The cyclic variety $\Gamma_d$ is defined by a set of vanishing polynomial cyclic conditions which relates to the critical loci. This variety will be then stratified and the critical loci will inherit a stratification. 
\begin{definition}[Determinantal, cyclic, and critical loci]
Define
\[
\Sigma_d^r:=\{\theta\in \Omega:\rk(\mu(\theta))=r\} \quad \text{\textit{Determinantal variety}},
\] 
\[
\Gamma_{d}
:=
\left\{
(\theta,W_0)\in \Omega \times \text{Mat}_{d_0 \times d_L}
\ \middle|\
W_{<\ell}W_{0}W_{>\ell}=0
\text{ for all }1\leq \ell\leq L
\right\}, \quad \text{\textit{Cyclic variety}}
\]
Let $\mathcal{F}^t:=\operatorname{Mat}_{d_0\times d_L}$ and let $\pi: \Gamma_d \to \mathcal{F} \times \mathcal{F}^t$ such that $\pi(\theta, W_0):= (\mu(\theta), W_0)$. Define the determinantal cyclic variety:  
\begin{comment}
Let $\iota_S:\Omega \longrightarrow \Omega \times \text{Mat}_{d_0 \times d_L}$ be an inclusion map such that $\iota_S(\theta):=(\theta, \Sigma_{XY}P_Q)$ and define:
\end{comment}
\[
\Gamma_{d}^r:=\Gamma_{d}\cap (\Sigma_d^r \times \text{Mat}_{d_0 \times d_L})
\]
Let $S$ be a subset of left-singular vectors of $M$, define the $S$-critical loci:
\[
\mathcal C^S_{d} = \{\theta\in\Omega | \ \nabla_{\theta} L(\theta)=0; \mu(\theta)= P_SW^{\star}\}
\]
\end{definition}
In proposition \ref{} we show that the $S$- critical loci is a fiber of $\pi$:
\[
\mathcal C^S_{d} \cong \Gamma^S_d:=\pi^{-1}(P_SW^{\star}, \Sigma_{XY}P_Q) \subset \Gamma^r_d
\]
\begin{definition}[Symmetry group]
For dimension vector $d=(d_0,...,d_L)$ define the symmetry group:
\[
G_d = \prod_{i=0}^{L} GL_{d_i}
\]
Consider the action of $G_d$:
\[
g.(\theta, W_0) = (g_L W_L g_{L-1}^{-1}, \dots, g_1 W_1 g_0^{-1}, g_0W_0g_L^{-1})
\]
and note that we have a well defined action of $G_d$ over the cyclic variety from $G_d$-invariance of the equations defining the cyclic variety. The variety $\Gamma_d^r$ also inherits an action of $G_d$ from the action of $G_d$ over $\Sigma_d^r \times \text{Mat}_{d_0 \times d_L}$, since the rank is invariant under action of $GL_d$. A consequence of the action is that we can stratify the variety $\Gamma^r_d$ into orbits (strata) $O_{\rho}$. In this paper we will show that the relevant geometric invariant $\rho$ parametrizing the orbits are the cyclic rank patterns.
\end{definition}
\begin{definition}\label{def:cyclic-rank-pattern}
For $\phi = (\theta,W_0)\in\Gamma_{d}$ we define its \emph{cyclic rank pattern} to be the array
\[
\rho(\phi)=(r_{ij}(\phi))_{0\le i,j\le L}
\quad\text{where}\quad
\rho_{ij}(\phi):=\mathrm{rank}\bigl(W_{j}...W_i\bigr) \quad \text{for} \quad \text{and} \quad \rho_{ii}(\phi) = d_i \text{and} \quad \rho_{i+1i}(\phi) = \mathrm{rank}\bigl(W_i\bigr)
\]
\end{definition}
\begin{theorem}\label{main-result}
Define the strata $\mathcal O^S_{\rho} := \mathcal O_{\rho} \cap \Gamma^S_{d}$ where  $\mathcal{O}_{\rho}$ is a $G_d$-orbit of the cyclic variety $\Gamma^r_{d}$.
Assuming that the data is non degenerate and the input covariance matrix is invertible then the loci of critical points of rank $r<r_\text{max}$ admits the following stratification parameterized by subset $S$ of singular vectors of the teacher with $r$ elements and admissible cyclic rank patterns $\rho\in R_d^{\mathrm{orb}}$:
\begin{align*}
  \mathcal{C}^r_d & = \bigsqcup_{|S|=r}\mathcal{C}^S_d = \bigsqcup_{|S|=r}\bigsqcup_{\substack{\rho\in R_d^{\mathrm{orb}}\\ \operatorname{rank}(\rho)=\operatorname{rank}(\Sigma_{XY}P_Q)}} \mathcal{O}^S_{\rho} \\
\end{align*}
\end{theorem}
\section{A stratification of the cyclic variety induces a stratification of the critical loci}
\label{sec:cyclic-stratification-critical-loci}

In this section we explain how the orbit stratification of the cyclic variety
induces a genuine orbit stratification of each fixed-\(S\) critical locus. All
closures are relative Zariski closures.

Fix a subset \(S\) of left singular vectors and set
\[
B_S:=P_SW^\star,
\qquad
E_S:=\Sigma_{XY}P_Q .
\]
Thus \(B_S\in\mathcal F=\operatorname{Mat}_{d_L\times d_0}\) and
\(E_S\in\mathcal F^t=\operatorname{Mat}_{d_0\times d_L}\). Recall that
\[
\pi:\Gamma_d\longrightarrow \mathcal F\times \mathcal F^t,
\qquad
\pi(\theta,W_0):=(\mu(\theta),W_0).
\]
We write
\[
\Gamma_d^S:=\pi^{-1}(B_S,E_S)
=
\{(\theta,W_0)\in\Gamma_d\mid \mu(\theta)=B_S,\ W_0=E_S\}.
\]
If \(\operatorname{rank}(B_S)=r\), then \(\Gamma_d^S\subseteq \Gamma_d^r\).

\begin{proposition}[The \(S\)-critical locus as a fiber]
\label{prop:S-critical-fiber}
Assume that \(\Sigma_X\) is invertible and that
\[
W^\star=\Sigma_{YX}\Sigma_X^{-1}.
\]
Then the map
\[
\Theta_S:\mathcal C_d^S\longrightarrow \Gamma_d^S,
\qquad
\Theta_S(\theta):=(\theta,E_S),
\]
is an isomorphism of varieties, with inverse given by projection onto the
\(\theta\)-coordinates. In particular,
\[
\mathcal C_d^S\cong \pi^{-1}(P_SW^\star,\Sigma_{XY}P_Q).
\]
\end{proposition}

\begin{proof}
Let \(\theta\in\Omega\) satisfy \(\mu(\theta)=B_S=P_SW^\star\). Then
\[
\Sigma_X(P_SW^\star)^\top-\Sigma_{XY}
=
-\Sigma_{XY}P_Q
=
-E_S.
\]
Hence, for every \(1\leq \ell\leq L\),
\[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
-2W_{<\ell}E_SW_{>\ell}.
\]
Therefore the gradient equations are equivalent, under the constraint
\(\mu(\theta)=B_S\), to the cyclic equations
\[
W_{<\ell}E_SW_{>\ell}=0
\qquad
\text{for every }1\leq \ell\leq L.
\]
These are precisely the equations saying that
\[
(\theta,E_S)\in \Gamma_d
\]
together with \(\pi(\theta,E_S)=(B_S,E_S)\). Thus
\[
\theta\in\mathcal C_d^S
\quad\Longleftrightarrow\quad
(\theta,E_S)\in\Gamma_d^S.
\]
The maps \(\theta\mapsto(\theta,E_S)\) and
\((\theta,E_S)\mapsto\theta\) are regular and inverse to one another.
\end{proof}

We now use the symmetry action. The group
\[
G_d=\prod_{i=0}^L GL_{d_i}
\]
acts on \(\Gamma_d\) by
\[
g\cdot(\theta,W_0)
=
(g_LW_Lg_{L-1}^{-1},\dots,g_1W_1g_0^{-1},g_0W_0g_L^{-1}).
\]
The map \(\pi\) is equivariant in the following sense:
\[
\pi(g\cdot(\theta,W_0))
=
(g_L\mu(\theta)g_0^{-1},\,g_0W_0g_L^{-1}).
\]
Define the stabilizer of the fixed pair \((B_S,E_S)\) by
\[
K_S
:=
\left\{
g\in G_d
\ \middle|\
g_LB_Sg_0^{-1}=B_S,\;
g_0E_Sg_L^{-1}=E_S
\right\}.
\]
Then \(K_S\) acts on the fiber \(\Gamma_d^S\).

Assume from now on that the rank-\(r\) cyclic variety has a finite
\(G_d\)-orbit stratification by cyclic rank patterns:
\[
\Gamma_d^r
=
\bigsqcup_{\rho\in R_d^r} O_\rho,
\]
where \(R_d^r\) denotes the finite set of cyclic rank patterns occurring in
\(\Gamma_d^r\). We assume moreover\footnote{Both of these assumptions will be proven later on in the paper} that the ambient cyclic orbit closures are
governed by the coordinatewise rank order:
\[
\overline{O_\rho}^{\,\Gamma_d^r}
=
\bigsqcup_{\substack{\rho'\in R_d^r\\ \rho'\leq_{\mathrm{rk}}\rho}}
O_{\rho'}.
\]
Here
\[
\rho'\leq_{\mathrm{rk}}\rho
\quad\Longleftrightarrow\quad
\rho'_{ij}\leq \rho_{ij}
\quad
\text{for every cyclic rank coordinate }(i,j).
\]

For each \(\rho\in R_d^r\), define
\[
\mathcal O_\rho^S
:=
O_\rho\cap\Gamma_d^S
=
O_\rho\cap\pi^{-1}(B_S,E_S).
\]
Let
\[
R_d^S
:=
\{\rho\in R_d^r\mid \mathcal O_\rho^S\neq\varnothing\}.
\]

\begin{proposition}[Induced \(K_S\)-orbit stratification of the \(S\)-critical locus]
\label{prop:critical-rank-pattern-stratification}
For every \(\rho\in R_d^S\), the stratum
\[
\mathcal O_\rho^S
=
O_\rho\cap\pi^{-1}(B_S,E_S)
\]
is a single \(K_S\)-orbit. Consequently,
\[
\Gamma_d^S
=
\bigsqcup_{\rho\in R_d^S}\mathcal O_\rho^S
\]
is a finite decomposition into smooth locally closed \(K_S\)-orbits. Via the
identification \(\mathcal C_d^S\cong\Gamma_d^S\) from
Proposition~\ref{prop:S-critical-fiber}, this gives a stratification of
\(\mathcal C_d^S\).

Define the fixed-\(S\) closure relation on \(R_d^S\) by
\[
\rho'\leq_S\rho
\quad\Longleftrightarrow\quad
\mathcal O_{\rho'}^S
\subseteq
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}.
\]
Equivalently, since the strata are \(K_S\)-orbits,
\[
\rho'\leq_S\rho
\quad\Longleftrightarrow\quad
\mathcal O_{\rho'}^S
\cap
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\neq\varnothing.
\]
Then the closure of each stratum is the union of the strata below it for this
relation:
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
=
\bigsqcup_{\substack{\rho'\in R_d^S\\ \rho'\leq_S\rho}}
\mathcal O_{\rho'}^S.
\]
Equivalently, if
\[
\rho'<_S\rho
\quad\Longleftrightarrow\quad
\rho'\leq_S\rho
\text{ and }
\rho'\neq\rho,
\]
then the frontier satisfies
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\setminus
\mathcal O_\rho^S
=
\bigsqcup_{\substack{\rho'\in R_d^S\\ \rho'<_S\rho}}
\mathcal O_{\rho'}^S.
\]

Moreover, the fixed-\(S\) closure order refines the ambient coordinatewise
rank order:
\[
\rho'\leq_S\rho
\quad\Longrightarrow\quad
\rho'\leq_{\mathrm{rk}}\rho.
\]
Thus
\[
\{\rho'\in R_d^S\mid \rho'\leq_S\rho\}
\subseteq
\{\rho'\in R_d^S\mid \rho'\leq_{\mathrm{rk}}\rho\}.
\]
The inclusion may be strict.
\end{proposition}

\begin{proof}
First we show that each nonempty \(\mathcal O_\rho^S\) is a single
\(K_S\)-orbit. Let
\[
x,y\in \mathcal O_\rho^S.
\]
Since \(x\) and \(y\) both lie in the \(G_d\)-orbit \(O_\rho\), there exists
\(g\in G_d\) such that
\[
y=g\cdot x.
\]
Because both \(x\) and \(y\) lie in the fiber \(\Gamma_d^S\), we have
\[
\pi(x)=(B_S,E_S),
\qquad
\pi(y)=(B_S,E_S).
\]
Using the equivariance of \(\pi\),
\[
(B_S,E_S)
=
\pi(y)
=
\pi(g\cdot x)
=
(g_LB_Sg_0^{-1},\,g_0E_Sg_L^{-1}).
\]
Hence \(g\in K_S\). Therefore \(y\in K_S\cdot x\). Conversely, \(K_S\)
preserves both the \(G_d\)-orbit \(O_\rho\) and the fiber \(\Gamma_d^S\), so
\[
K_S\cdot x\subseteq O_\rho\cap\Gamma_d^S.
\]
Thus
\[
\mathcal O_\rho^S=K_S\cdot x.
\]

Since algebraic group orbits are smooth and locally closed, each
\(\mathcal O_\rho^S\) is smooth and locally closed. The decomposition
\[
\Gamma_d^S
=
\bigsqcup_{\rho\in R_d^S}\mathcal O_\rho^S
\]
follows by intersecting the disjoint orbit decomposition of \(\Gamma_d^r\) with
the fiber \(\Gamma_d^S\) and discarding the empty intersections.

We now prove the frontier condition. Fix \(\rho\in R_d^S\) and set
\[
Z_\rho
:=
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}.
\]
Because \(\mathcal O_\rho^S\) is \(K_S\)-stable, its closure \(Z_\rho\) is also
\(K_S\)-stable. Suppose
\[
Z_\rho\cap \mathcal O_{\rho'}^S\neq\varnothing.
\]
Choose \(x\in Z_\rho\cap \mathcal O_{\rho'}^S\). Since
\(\mathcal O_{\rho'}^S\) is a \(K_S\)-orbit,
\[
\mathcal O_{\rho'}^S=K_S\cdot x.
\]
Since \(Z_\rho\) is \(K_S\)-stable,
\[
K_S\cdot x\subseteq Z_\rho.
\]
Therefore
\[
\mathcal O_{\rho'}^S\subseteq Z_\rho.
\]
This proves the frontier condition. The closure equality follows because the
\(\mathcal O_{\rho'}^S\)'s form a finite disjoint decomposition of
\(\Gamma_d^S\).

It remains to compare \(\leq_S\) with the coordinatewise rank order. Since
\[
\mathcal O_\rho^S\subseteq O_\rho
\qquad
\text{and}
\qquad
\Gamma_d^S\subseteq \Gamma_d^r,
\]
we have
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\subseteq
\Gamma_d^S\cap \overline{O_\rho}^{\,\Gamma_d^r}.
\]
Using the ambient cyclic closure formula,
\[
\overline{O_\rho}^{\,\Gamma_d^r}
=
\bigsqcup_{\substack{\rho''\in R_d^r\\ \rho''\leq_{\mathrm{rk}}\rho}}
O_{\rho''},
\]
we obtain
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\subseteq
\Gamma_d^S
\cap
\bigsqcup_{\substack{\rho''\in R_d^r\\ \rho''\leq_{\mathrm{rk}}\rho}}
O_{\rho''}
=
\bigsqcup_{\substack{\rho''\in R_d^S\\ \rho''\leq_{\mathrm{rk}}\rho}}
\mathcal O_{\rho''}^S.
\]
Therefore, if \(\rho'\leq_S\rho\), then
\[
\mathcal O_{\rho'}^S
\subseteq
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\]
forces
\[
\rho'\leq_{\mathrm{rk}}\rho.
\]
This proves
\[
\leq_S\ \subseteq\ \leq_{\mathrm{rk}}
\]
on the set \(R_d^S\).
\end{proof}

\begin{remark}
The relation \(\leq_S\) is the actual frontier relation on the fixed-\(S\)
critical locus. The coordinatewise order \(\leq_{\mathrm{rk}}\) is only an
ambient necessary condition for specialization. Thus one should not replace
\(\leq_S\) by \(\leq_{\mathrm{rk}}\) in the closure formula.

Indeed, even if
\[
\rho'\leq_{\mathrm{rk}}\rho,
\]
the intersection
\[
O_{\rho'}\cap\pi^{-1}(B_S,E_S)
\]
may be empty. In that case there is no \(K_S\)-orbit
\(\mathcal O_{\rho'}^S\) in the fixed-\(S\) critical locus.
\end{remark}
In the following sections we will show how to stratify $\Gamma_d^r$ using quiver representation theory.
\section{The cyclic rank patterns as geometric invariants of deep linear networks}
Goal: Stratify the cyclic variety with cyclic rank patterns.
\subsection{Essentials of Quiver representation theory}
A quiver is a graph $(I,E)$ with vertices $I$, edges $E$ and an orientation. A quiver representation assigns a vector space to $V_i \cong \mathbb{R}^{d_i}$ and a linear map $\phi: V_i \to V_{i+1}$ with vector dimension $d_i$. 
\subsection{The cyclic Nakayama algebra}

Throughout this subsection, put \(n=L+1\), and take all indices modulo
\(n\). Unless otherwise stated, all \(A\)-modules are finite-dimensional
left \(A\)-modules.

Let \(Q=C_{L+1}\) be the cyclic quiver with vertex set
\[
Q_0=\{0,\ldots,L\}
\]
and arrows
\[
a_l:l-1\longrightarrow l,
\qquad l\in \{0,\ldots,L\},
\]
where \(a_0:L\to 0\). We read products of arrows from right to left. For
\(1\leq l\leq L\), define the length \(L\) path
\[
p_l:=a_{l-1}\cdots a_1a_0a_L\cdots a_{l+1}.
\]
Thus \(p_l\) is the unique path of length \(L\) starting at \(l\), ending
at \(l-1\), and passing through the closing arrow \(a_0\).

Let
\[
I=\langle p_1,\ldots,p_L\rangle
\]
and define the bound path algebra
\[
A:=kQ/I.
\]
Equivalently, an \(A\)-module with dimension vector
\(d=(d_0,\ldots,d_L)\) is a representation of \(Q\) on vector spaces
\(V_i\simeq k^{d_i}\), with linear maps
\[
\phi_l:V_{l-1}\longrightarrow V_l,
\]
satisfying the relations
\[
\phi_{l-1}\circ\cdots\circ \phi_1\circ \phi_0\circ \phi_L
\circ\cdots\circ \phi_{l+1}=0,
\qquad 1\leq l\leq L.
\]
We write \(\Rep_d(A)\) for the corresponding representation variety.

\begin{proposition}
For \(k=\mathbb R\), the cyclic variety \(\Gamma_d\) is identified with
\(\Rep_d(A)\) by
\[
W_l=\phi_l,
\qquad
V_i\simeq \mathbb R^{d_i}.
\]
\end{proposition}

\begin{proof}
Under this identification, the defining equations of \(\Gamma_d\) are
exactly the relations generated by the paths \(p_1,\ldots,p_L\). Hence
\(\Gamma_d\cong \Rep_d(A)\).
\end{proof}

Let \(R\subset kQ\) be the arrow ideal, i.e. the ideal generated by all
arrows. Every path of length \(L+1\) contains one of the paths
\(p_1,\ldots,p_L\) as a subpath. Therefore
\[
R^{L+1}\subset I.
\]
It follows that \(A\) is finite-dimensional. Moreover,
\[
J(A)=R/I,
\]
so \(J(A)\) is spanned by the nonzero paths of positive length, and
\[
A/J(A)\cong \bigoplus_{i=0}^L k e_i.
\]
In particular, the simple \(A\)-modules are
\[
S_i:=P_i/JP_i,
\qquad
P_i:=Ae_i.
\]

\begin{definition}
A finite-length module \(U\) is called \emph{uniserial} if its submodules
are totally ordered by inclusion. Equivalently, \(U\) has a unique
composition series.

A finite-dimensional algebra is called a Nakayama algebra if its
indecomposable projective and indecomposable injective modules are
uniserial.
\end{definition}

\begin{proposition}
The algebra \(A=kQ/I\) is a finite-dimensional Nakayama algebra. Its
Kupisch series is
\[
(c_0,\ldots,c_L)=(L+1,L,\ldots,L).
\]
\end{proposition}

\begin{proof}
We first describe the indecomposable projectives. With our convention for
path multiplication, \(P_i=Ae_i\) is spanned by the nonzero paths starting
at \(i\). Let
\[
q_{i,0}:=e_i,
\qquad
q_{i,\ell}:=a_{i+\ell}\cdots a_{i+1}
\]
be the unique path of length \(\ell\) starting at \(i\). The path \(q_{i,\ell}\)
is nonzero precisely for
\[
0\leq r<c_i,
\]
where
\[
c_0=L+1,
\qquad
c_i=L \quad \text{for } 1\leq i\leq L.
\]
Indeed, the path starting at \(0\) of length \(L\) does not pass through
\(a_0\), so it is nonzero, whereas the path starting at \(0\) of length
\(L+1\) vanishes. If \(i\neq 0\), then the path of length \(L\) starting
at \(i\) is exactly the relation \(p_i\), so it vanishes.

Thus
\[
P_i=\langle q_{i,0},q_{i,1},\ldots,q_{i,c_i-1}\rangle_k,
\]
and
\[
J^sP_i
=
\langle q_{i,s},q_{i,s+1},\ldots,q_{i,c_i-1}\rangle_k,
\qquad
0\leq s\leq c_i.
\]
In particular,
\[
P_i\supset JP_i\supset J^2P_i\supset\cdots\supset J^{c_i}P_i=0.
\]
Moreover,
\[
J^sP_i/J^{s+1}P_i\cong S_{i+s}.
\]
Hence this chain is a composition series.

It remains to check that it is the unique composition series. Let
\(N\subset P_i\) be a nonzero submodule. Since the idempotents \(e_v\)
act as projections and each space \(e_vP_i\) is at most one-dimensional,
\(N\) is spanned by some subset of the basis elements \(q_{i,\ell}\). If
\(q_{i,r}\in N\), then stability under the action of the arrows implies
\[
q_{i,\ell+1},q_{i,\ell+2},\ldots,q_{i,c_i-1}\in N.
\]
Therefore
\[
N=J^sP_i
\]
for some \(s\). Hence the submodules of \(P_i\) are totally ordered, and
\(P_i\) is uniserial.

The modules \(P_i=Ae_i\) form a complete set of indecomposable
projective \(A\)-modules. The same argument applied to the opposite
algebra shows that the indecomposable injective \(A\)-modules are also
uniserial. Therefore \(A\) is a finite-dimensional Nakayama algebra.
\end{proof}

\begin{lemma}
Let \(B\) be a finite-dimensional Nakayama algebra. Then every
indecomposable finite-dimensional \(B\)-module is uniserial.
\end{lemma}

\begin{proof}
This comes from the Artinian characterization of Nakayama algebras:
for a finite-dimensional algebra \(B\), the following are equivalent:
\begin{enumerate}
  \item \(B\) is Nakayama;
  \item every indecomposable finite-dimensional \(B\)-module is
  uniserial;
  \item every finite-dimensional \(B\)-module is a direct sum of
  uniserial modules.
\end{enumerate}
Since \(B\) is finite-dimensional, it is Artinian, so this
characterization applies.
\end{proof}
\begin{definition}
For \(i,j\in Q_0\), let
\[
\delta(i,j)\in \{0,\ldots,L\}
\]
be the unique integer such that
\[
j\equiv i+\delta(i,j)\pmod {L+1}.
\]
Whenever
\[
\delta(i,j)+1\leq c_i,
\]
define the interval module \(M(i,j)\) by
\[
M(i,j)_v=
\begin{cases}
k, & \text{if } v\equiv i+\ell \pmod {L+1}
      \text{ for all }0\leq \ell\leq \delta(i,j),\\
0, & \text{otherwise,}
\end{cases}
\]
and
\[
M(i,j)_{a_m}=
\begin{cases}
\operatorname{id}_k,
& \text{if } m\equiv i+\ell \pmod {L+1}
  \text{ for all }1\leq \ell\leq \delta(i,j),\\
0, & \text{otherwise.}
\end{cases}
\]
The condition \(\delta(i,j)+1\leq c_i\) is exactly the condition that the
relations in \(I\) are satisfied.
\end{definition}
\begin{theorem}
Every nonzero indecomposable finite-dimensional \(A\)-module is
isomorphic to a unique interval module \(M(i,j)\) with
\[
\delta(i,j)+1\leq c_i.
\]
Equivalently, the indecomposable \(A\)-modules are given by:
\[
M(i,i+\ell-1),
\qquad
0\leq i\leq L,\quad 1\leq \ell\leq c_i,
\]
where indices are taken modulo \(L+1\).
\end{theorem}

\begin{proof}
Let \(M\) be a nonzero indecomposable finite-dimensional \(A\)-module.
Since \(A\) is a finite-dimensional Nakayama algebra, the previous lemma
implies that \(M\) is uniserial. Hence \(M\) has a unique maximal
submodule, and therefore its top is simple:
\[
\operatorname{top}M:=M/JM\cong S_i
\]
for a unique vertex \(i\).

Let
\[
\pi:P_i\twoheadrightarrow M
\]
be the projective cover of \(M\). Such a projective cover exists because
\(A\) is finite-dimensional, and it must be \(P_i\) because
\(\operatorname{top}P_i\cong S_i\). Let \(K=\ker \pi\). Then
\[
M\cong P_i/K.
\]
Since \(P_i\) is uniserial, its submodules are precisely
\[
P_i,\ JP_i,\ J^2P_i,\ldots,\ J^{c_i}P_i=0.
\]
Because \(M\neq 0\), the kernel is not all of \(P_i\). Thus there is a
unique integer \(\ell\) with \(1\leq \ell\leq c_i\) such that
\[
K=J^{\ell}P_i.
\]
Therefore
\[
M\cong P_i/J^{\ell}P_i.
\]

As a representation of the quiver, \(P_i/J^lP_i\) has one-dimensional
vector spaces at the vertices
\[
i,\ i+1,\ \ldots,\ i+\ell-1
\]
and zero vector spaces elsewhere. The arrow maps along this interval are
\(\operatorname{id}_k\), and all other arrow maps are zero. Hence
\[
P_i/J^{\ell}P_i\cong M(i,i+\ell-1).
\]

Conversely, for every \(i\) and every \(1\leq r\leq c_i\), the interval
module \(M(i,i+\ell-1)\) is isomorphic to \(P_i/J^{\ell}P_i\). Since it is a
nonzero quotient of a uniserial module, it is uniserial, hence
indecomposable.

Finally, uniqueness follows because the top of \(M(i,i+\ell-1)\) is
\(S_i\), which determines \(i\), and its composition length is \(\ell\),
which determines the endpoint \(i+\ell-1\). This proves the classification.
\end{proof}

\begin{definition}
Let
\[
c_0=L+1,
\qquad
c_i=L \quad \text{for }1\leq i\leq L.
\]
Define the set of admissible intervals by
\[
\mathcal I
:=
\{(i,\ell)\mid i\in \{0,\ldots,L\},\ 1\leq \ell\leq c_i\}.
\]
For \((i,\ell)\in \mathcal I\), write
\[
j(i,\ell):=i+\ell-1 \pmod {L+1}.
\]
The corresponding interval module is
\[
M_{\ell}(i):=M(i,j(i,\ell)).
\]
Equivalently, \(M_{\ell}(i)\) is supported on the nonzero path
\[
i\to i+1\to \cdots \to i+\ell-1,
\]
with one-dimensional vector spaces along this path and identity maps
along the arrows of the path.

The number of indecomposables is
\[
|\mathcal I|
=
c_0+\sum_{i=1}^L c_i
=
(L+1)+L^2
=
L^2+L+1.
\]
\end{definition}

\begin{corollary}[Krull--Schmidt decomposition for \(\Rep_d(A)\)]
Every finite-dimensional \(A\)-module \(M\) admits a decomposition
\[
M
\cong
\bigoplus_{(i,\ell)\in \mathcal I}
M_{\ell}(i)^{\oplus m_{i,\ell}},
\]
where \(m_{i,\ell}\in \mathbb Z_{\geq 0}\), and only finitely many
multiplicities are nonzero.

Moreover, the multiplicities \(m_{i,\ell}\) are uniquely determined by
\(M\). Equivalently, the decomposition is unique up to permutation and
isomorphism of the indecomposable summands.
\end{corollary}

\begin{proof}
Since \(A\) is finite-dimensional, the category of finite-dimensional
\(A\)-modules is a Krull--Schmidt category. Hence every finite-dimensional
\(A\)-module decomposes as a finite direct sum of indecomposable
\(A\)-modules, and this decomposition is unique up to permutation and
isomorphism of the indecomposable summands.

By the classification theorem, every nonzero indecomposable
\(A\)-module is isomorphic to an interval module
\[
M_{\ell}(i)\cong P_i/J^{\ell}P_i
\]
for a unique pair
\[
(i,\ell)\in \mathcal I.
\]
Therefore every finite-dimensional \(A\)-module has a decomposition of
the form
\[
M
\cong
\bigoplus_{(i,\ell)\in \mathcal I}
M_{\ell}(i)^{\oplus m_{i,\ell}}.
\]
The uniqueness statement in Krull--Schmidt implies that the
multiplicities \(m_{i,\ell}\) are uniquely determined.
\end{proof}
\subsection{From multiplicity patterns to rank patterns}
\begin{definition}[Multiplicity Patterns]
    Consider a $A$-module $M$ of vector dimension $d$ and its Krull-Schmidt decomposition into $M_{\ell}(i)$ for $(i,\ell)\in \mathcal I$. For each vertex $k$ the dimension $d_k$ of the representation $M$ at vertex $k$ satisfies the dimension constraint:
\[d_k \;=\;
\sum_{(i,\ell)\in \mathcal I}
m_{i,\ell}\,\mathbf{1}\!\left\{
k \in \{i,i+1,\dots,i+\ell\} \ (\mathrm{mod}\ L+1)
\right\}\quad k=0,\dots,L\]
Define the set of \emph{multiplicity patterns} as
\[\mathcal M_d^+ := \left\{(m_{i,\ell})\in \mathbb{Z}_{\ge 0}^{\mathcal I}\middle|\ d_k \;=\;
\sum_{(i,\ell)\in \mathcal I}
m_{i,\ell}\,\mathbf{1}\!\left\{
k \in \{i,i+1,\dots,i+\ell\} \ (\mathrm{mod}\ L+1)
\right\}\quad k=0,\dots,L\right\}.\]
\end{definition}
We want to stratify the critical loci in terms of the rank patterns as they are more intuitive and native to the DLN representation. In this section we show that there is a bijection between the multiplicity patterns and a set of cyclic rank patterns satisfying some positivity constraint. 
\begin{definition}
For \((a,b)\in \mathcal I\), let
\[
q_{ab}
\]
denote the unique nonzero oriented path of \(A\) over the interval \([a,b]\) from vertex $a$ to vertex $b$. 
\end{definition}

\begin{definition}[Cyclic rank pattern]
Let \(M\) be an \(A\)-module of dimension vector \(d\). Its
lifted cyclic rank pattern is the array
\[
\rho(M)=(\rho_{ab}(M))_{(a,b)\in \mathcal I},
\qquad
\rho_{ab}(M):=\rk\bigl(q_{ab}(M)\bigr).
\]
\end{definition}

Two coordinates play a distinguished role:
\[
\rho_{0L}(M)=\rk(q_{0L}(M)),
\qquad
\rho_{L,0}(M)=\rk(q_{L,0}(M)).
\]
In the DLN representation these are exactly the end-to-end rank and the rank of the
closing arrow.

\begin{proposition}[Forward formula]
\label{prop:forward}
Let \(M\) have multiplicity pattern
\[
m=(m_{uv})_{(u,v)\in \mathcal I}.
\]
Then for every \((a,b)\in \mathcal I_L^{\mathrm{wt}}\),
\[
\rho_{ab}(M)
=
\sum_{\substack{(u,v)\in \mathcal I\\ u\leq a\leq b\leq v}}
m_{uv}.
\]
\end{proposition}

\begin{proof}
On an indecomposable interval module \(M(u,v)\), the path \(q_{ab}\) acts with
rank \(1\) exactly when the path interval \([a,b]\) is contained in \([u,v]\),
and otherwise with rank \(0\). The claim follows by additivity of rank on
direct sums.
\end{proof}

\begin{proposition}[Inversion formula]
\label{prop:inversion}
Extend every rank pattern by the convention
\[
\rho_{uv}=0
\qquad
\text{for }(u,v)\notin \mathcal I.
\]
Then for every admissible interval \((a,b)\in \mathcal I\),
\[
m_{ab}
=
\rho_{ab}-\rho_{a-1,b}-\rho_{a,b+1}+\rho_{a-1,b+1}.
\]
\end{proposition}

\begin{proof}
Insert the forward formula into the right-hand side and note that, for every
fixed interval \((u,v)\), the coefficient of \(m_{uv}\) is
\[
\mathbf 1_{u\leq a\leq b\leq v}
-\mathbf 1_{u\leq a-1\leq b\leq v}
-\mathbf 1_{u\leq a\leq b+1\leq v}
+\mathbf 1_{u\leq a-1\leq b+1\leq v}.
\]
This coefficient equals \(1\) precisely when \((u,v)=(a,b)\) and equals \(0\)
otherwise. A more detailed telescoping proof is recorded in
Appendix~\ref{app:inversion}.
\end{proof}

\begin{definition}[Admissible rank-pattern set]
\[
\mathcal R_d
:=
\left\{
\rho\in \Z_{\geq 0}^{\mathcal I}
\ \middle|\
\begin{array}{l}
\rho_{aa}=d_a \text{ for every }0\leq a\leq L,\\[2pt]
\rho_{ab}-\rho_{a-1,b}-\rho_{a,b+1}+\rho_{a-1,b+1}\geq 0\\
\hspace{2.7cm}\text{for every }(a,b)\in \mathcal I
\end{array}
\right\}.
\]
\end{definition}

\begin{proposition}[Multiplicity-rank bijection]
\label{thm:rank-bijection}
The multiplicity-pattern set and the rank-pattern set are canonically in
bijection:
\[
\mathcal M_d^+
\xrightarrow{\sim}
\mathcal R_d.
\]
\end{proposition}

\begin{proof}
The forward map sends a multiplicity pattern to the rank pattern given by
Proposition~\ref{prop:forward}. The inverse map is the finite-difference formula
of Proposition~\ref{prop:inversion}. The diagonal condition \(\rho_{aa}=d_a\)
ensures that the resulting multiplicity pattern has dimension vector \(d\).
\end{proof}
\section{Stratification of the critical loci with cyclic rank patterns}
\label{sec:stratification-critical-loci-rank-patterns}

We have identified the cyclic variety \(\Gamma_d\) with the representation
variety of the Nakayama algebra \(A=kQ/I\) with dimension vector \(d\). Under
this identification, the \(G_d\)-orbits in \(\Gamma_d\) are precisely the
isomorphism classes of \(A\)-modules with dimension vector \(d\). By the
Krull--Schmidt theorem, each such module has a unique decomposition into
indecomposable Nakayama modules, and hence each orbit is indexed by a
multiplicity pattern \(m\in\mathcal M_d^+\).

\begin{proposition}[Orbit stratification of \(\Gamma_d\) by indecomposable Nakayama modules]
\label{prop:Gamma_mult_orbit_stratification}
Fix \(L\geq 1\), let \(Q=C_{L+1}\) be the oriented cyclic quiver on vertices
\(\{0,\dots,L\}\), and let \(A=kQ/I\) be the Nakayama quiver algebra defined
above. Then
\[
\Gamma_d
=
\bigsqcup_{m\in\mathcal M_d^+}\mathcal O_m
=
\bigsqcup_{m\in\mathcal M_d^+}G_d\cdot \phi_m,
\]
where \(\phi_m\) is any representative whose associated \(A\)-module has
multiplicity pattern \(m\). In particular, the \(G_d\)-orbits in \(\Gamma_d\)
are parametrized by the multiplicity data of indecomposable \(A\)-modules.
\end{proposition}

Using the multiplicity-rank bijection of
Proposition~\ref{thm:rank-bijection}, we may equivalently index the same
orbits by cyclic rank patterns. Thus we write
\[
\Gamma_d
=
\bigsqcup_{\rho\in\mathcal R_d}\mathcal O_\rho,
\]
where \(\mathcal O_\rho\) is the \(G_d\)-orbit with cyclic rank pattern
\(\rho\). We use the coordinatewise rank order
\[
\rho'\leq_{\mathrm{rk}}\rho
\quad\Longleftrightarrow\quad
\rho'_{ab}\leq \rho_{ab}
\quad
\text{for every cyclic rank coordinate }(a,b)\in\mathcal I.
\]
The orbit stratification of the cyclic variety satisfies the closure formula
\[
\overline{\mathcal O_\rho}^{\,\Gamma_d}
=
\bigsqcup_{\substack{\rho'\in\mathcal R_d\\
\rho'\leq_{\mathrm{rk}}\rho}}
\mathcal O_{\rho'}.
\]
Equivalently, the frontier of \(\mathcal O_\rho\) in \(\Gamma_d\) is the union
of the strictly lower coordinatewise rank-pattern orbits.

We now specialize this stratification to a fixed \(S\)-critical locus. Put
\[
B_S:=P_SW^\star,
\qquad
E_S:=\Sigma_{XY}P_Q,
\]
and recall that
\[
\Gamma_d^S
:=
\pi^{-1}(B_S,E_S)
\subseteq \Gamma_d.
\]
By Proposition~\ref{prop:S-critical-fiber}, the map
\[
\Theta_S:\mathcal C_d^S\longrightarrow \Gamma_d^S,
\qquad
\Theta_S(\theta)=(\theta,E_S),
\]
identifies the \(S\)-critical locus with the fiber \(\Gamma_d^S\).

Let
\[
r:=\operatorname{rank}(B_S),
\qquad
q_S:=\operatorname{rank}(E_S).
\]
Under the nondegeneracy assumptions of the main theorem, \(r=|S|\) and
\(q_S=d_L-r\). The two distinguished cyclic rank coordinates satisfy
\[
\rho_{0L}=\operatorname{rank}(\mu(\theta)),
\qquad
\rho_{L0}=\operatorname{rank}(W_0).
\]
Hence every point of \(\Gamma_d^S\) has
\[
\rho_{0L}=r,
\qquad
\rho_{L0}=q_S.
\]
These rank conditions are necessary but not sufficient for a cyclic orbit to
meet the fixed fiber \(\Gamma_d^S\).

For each \(\rho\in\mathcal R_d\), define the induced fixed-\(S\) stratum
\[
\mathcal O_\rho^S
:=
\mathcal O_\rho\cap \Gamma_d^S
=
\mathcal O_\rho\cap \pi^{-1}(B_S,E_S).
\]
We keep only the nonempty intersections and define
\[
\mathcal R_d^S
:=
\left\{
\rho\in\mathcal R_d
\ \middle|\
\rho_{0L}=r,\;
\rho_{L0}=q_S,\;
\mathcal O_\rho\cap\Gamma_d^S\neq\varnothing
\right\}.
\]
Thus
\[
\Gamma_d^S
=
\bigsqcup_{\rho\in\mathcal R_d^S}\mathcal O_\rho^S.
\]

The relevant symmetry group for the fixed fiber is the stabilizer of the pair
\((B_S,E_S)\). Define
\[
K_S
:=
\left\{
g\in G_d
\ \middle|\
g_LB_Sg_0^{-1}=B_S,\;
g_0E_Sg_L^{-1}=E_S
\right\}.
\]
The equivariance of \(\pi\),
\[
\pi(g\cdot(\theta,W_0))
=
(g_L\mu(\theta)g_0^{-1},\,g_0W_0g_L^{-1}),
\]
shows that \(K_S\) acts on \(\Gamma_d^S\).

\begin{theorem}[Induced orbit stratification of the \(S\)-critical locus]
\label{prop:critical-rank-pattern-decomposition}
For every \(\rho\in\mathcal R_d^S\), the stratum
\[
\mathcal O_\rho^S
=
\mathcal O_\rho\cap\Gamma_d^S
\]
is a single \(K_S\)-orbit. Consequently, from proposition \ref{prop:critical-rank-pattern-stratification}
\[
\Gamma_d^S
=
\bigsqcup_{\rho\in\mathcal R_d^S}\mathcal O_\rho^S
\]
is a finite stratification into smooth locally closed \(K_S\)-orbits. Define the fixed-\(S\) closure relation by
\[
\rho'\leq_S\rho
\quad\Longleftrightarrow\quad
\mathcal O_{\rho'}^S
\subseteq
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}.
\]
Equivalently, since the strata are \(K_S\)-orbits,
\[
\rho'\leq_S\rho
\quad\Longleftrightarrow\quad
\mathcal O_{\rho'}^S
\cap
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\neq\varnothing.
\]
Then
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
=
\bigsqcup_{\substack{\rho'\in\mathcal R_d^S\\
\rho'\leq_S\rho}}
\mathcal O_{\rho'}^S.
\]
Equivalently, if
\[
\rho'<_S\rho
\quad\Longleftrightarrow\quad
\rho'\leq_S\rho
\text{ and }
\rho'\neq \rho,
\]
then the frontier satisfies
\[
\overline{\mathcal O_\rho^S}^{\,\Gamma_d^S}
\setminus
\mathcal O_\rho^S
=
\bigsqcup_{\substack{\rho'\in\mathcal R_d^S\\
\rho'<_S\rho}}
\mathcal O_{\rho'}^S.
\]

Moreover, the fixed-\(S\) closure order refines the ambient coordinatewise rank
order:
\[
\rho'\leq_S\rho
\quad\Longrightarrow\quad
\rho'\leq_{\mathrm{rk}}\rho.
\]
Thus
\[
\{\rho'\in\mathcal R_d^S\mid \rho'\leq_S\rho\}
\subseteq
\{\rho'\in\mathcal R_d^S\mid \rho'\leq_{\mathrm{rk}}\rho\}.
\]
The inclusion may be strict. Via the identification \(\mathcal C_d^S\cong\Gamma_d^S\), define
\[
\mathcal C_\rho^S
:=
\Theta_S^{-1}(\mathcal O_\rho^S).
\]
Then
\[
\mathcal C_d^S
=
\bigsqcup_{\rho\in\mathcal R_d^S}\mathcal C_\rho^S.
\]
Consequently, we have a stratification of the variety of the critical points:
\[
\mathcal C_d
=
\bigsqcup_r \mathcal C_d^r
=
\bigsqcup_r \bigsqcup_{|S|=r}\mathcal C_d^S
=
\bigsqcup_r \bigsqcup_{|S|=r}
\bigsqcup_{\rho\in\mathcal R_d^S}
\mathcal C_\rho^S.
\]
Equivalently, after identifying \(\mathcal C_d^S\) with \(\Gamma_d^S\), the last
strata may be denoted by the corresponding \(K_S\)-orbits
\(\mathcal O_\rho^S\).
\end{theorem}
\begin{remark}[Compatibility of the fixed product fiber and choice of base field]
\label{rem:base-field-and-fixed-fiber}
Set
\[
B_S:=P_SW^\ast,
\qquad
E_S:=\Sigma_{XY}P_Q.
\]
Then
\[
E_SB_S
=
\Sigma_{XY}P_QP_SW^\ast
=
0
\]
because \(P_QP_S=0\). However, the other composition
\[
B_SE_S
=
P_SW^\ast\Sigma_{XY}P_Q
\]
does not vanish in general. It is an additional compatibility
condition.

This compatibility is necessary for the fixed product fiber
\[
\pi^{-1}(B_S,E_S)
\subseteq \Gamma_d
\]
to be nonempty. Indeed, if
\[
(\theta,W_0)\in \Gamma_d,
\qquad
\mu(\theta)=W_L\cdots W_1,
\]
then the cyclic relations imply
\[
\mu(\theta)W_0=0,
\qquad
W_0\mu(\theta)=0.
\]
For example, the relation with \(\ell=L\) gives
\[
W_{L-1}\cdots W_1W_0=0,
\]
and hence
\[
\mu(\theta)W_0
=
W_L(W_{L-1}\cdots W_1W_0)
=
0.
\]
Similarly, the relation with \(\ell=1\) gives
\[
W_0W_L\cdots W_2=0,
\]
and hence
\[
W_0\mu(\theta)
=
(W_0W_L\cdots W_2)W_1
=
0.
\]
Therefore, if
\[
\pi(\theta,W_0)=(B_S,E_S),
\]
then necessarily
\[
B_SE_S=0,
\qquad
E_SB_S=0.
\]
Consequently, if \(B_SE_S\neq 0\), then
\[
\pi^{-1}(B_S,E_S)=\varnothing,
\]
and the corresponding \(S\)-critical locus is empty.

In our deep linear network setting, this condition is
satisfied when \(P_S\) and \(P_Q\) are the spectral projectors associated with
the left singular directions of the teacher matrix
\[
M:=\Sigma_{YX}\Sigma_X^{-1/2}.
\]
Indeed,
\[
W^\ast\Sigma_{XY}
=
\Sigma_{YX}\Sigma_X^{-1}\Sigma_{XY}
=
MM^\top.
\]
Hence, if \(P_S\) and \(P_Q\) are complementary spectral projectors for
\(MM^\top\), then
\[
B_SE_S
=
P_SW^\ast\Sigma_{XY}P_Q
=
P_SMM^\top P_Q
=
0.
\]
If instead \(P_S\) is defined using the left singular vectors of \(W^\ast\)
itself, then \(B_SE_S=0\) need not hold unless, for example, \(\Sigma_X=I\).

We now discuss the base field. The representation-theoretic constructions are
valid over both \(k=\mathbb R\) and \(k=\mathbb C\). The cyclic variety
\(\Gamma_{d,k}\), the group \(G_{d,k}\), and the orbit strata
\(\mathcal O_{\rho,k}\) are defined over \(k\). The orbit decomposition is
field-independent: the indecomposable Nakayama modules have representatives
with matrices whose entries lie in \(\{0,1\}\). Thus the same cyclic rank
patterns occur over \(\mathbb R\) and over \(\mathbb C\), and the
\(k\)-rational points of each nonempty orbit are Zariski dense in that orbit.

Assume now that the fixed pair \((B_S,E_S)\) is compatible, i.e.
\[
B_SE_S=0,
\qquad
E_SB_S=0.
\]
Let
\[
r:=\operatorname{rank}(B_S),
\qquad
q_S:=\operatorname{rank}(E_S).
\]
The pair \((B_S,E_S)\) is then a representation of the two-vertex
radical-square-zero cyclic quiver
\[
V_0 \rightleftarrows V_L,
\qquad
BE=0,\quad EB=0.
\]
Such pairs are classified, over either \(\mathbb R\) or \(\mathbb C\), by the
two ranks \(r\) and \(q_S\). Equivalently, after choosing bases, the pair is
conjugate to the normal form
\[
B_{r,q_S}(u,v,w)=(u,0,0),
\qquad
E_{r,q_S}(a,b,c)=(0,b,0),
\]
with respect to decompositions
\[
k^{d_0}\cong k^r\oplus k^{q_S}\oplus k^{d_0-r-q_S},
\qquad
k^{d_L}\cong k^r\oplus k^{q_S}\oplus k^{d_L-r-q_S},
\]
whenever these dimensions are nonnegative. This normal form is defined over
\(\{0,1\}\), so the corresponding real and complex orbits have the same
algebraic geometry.

Therefore, for the fixed fiber
\[
\Gamma_{d,k}^S:=\pi_k^{-1}(B_S,E_S),
\]
the induced strata
\[
\mathcal O_{\rho,k}^S
:=
\mathcal O_{\rho,k}\cap \Gamma_{d,k}^S
\]
are independent of the choice \(k=\mathbb R\) or \(k=\mathbb C\), provided
all objects are taken over the same base field. More precisely,
\[
\mathcal O_{\rho,k}^S\neq\varnothing
\]
if and only if the outer cyclic rank coordinates satisfy
\[
\rho_{0L}=r,
\qquad
\rho_{L0}=q_S.
\]
Equivalently,
\[
\mathcal R_d^S
=
\{\rho\in\mathcal R_d\mid
\rho_{0L}=r,\ \rho_{L0}=q_S\}.
\]
If one does not want to assume the compatibility condition explicitly, the
safe definition is instead
\[
\mathcal R_d^S
:=
\{\rho\in\mathcal R_d\mid
\mathcal O_\rho\cap \pi^{-1}(B_S,E_S)\neq\varnothing\}.
\]

For every nonempty induced stratum, the stabilizer
\[
K_S
=
\left\{
g\in G_d
\ \middle|\
g_LB_Sg_0^{-1}=B_S,\;
g_0E_Sg_L^{-1}=E_S
\right\}
\]
acts transitively:
\[
\mathcal O_\rho^S
=
K_S\cdot x
\]
for any \(x\in\mathcal O_\rho^S\). Hence the \(K_S\)-orbit stratification and
its Zariski frontier relation are the same over \(\mathbb R\) and
\(\mathbb C\), after complexification.

The only distinction we do not identify across base fields is the ordinary
real analytic connected-component decomposition. A real locus of a complex
algebraic orbit may have more than one Euclidean connected component. This
does not affect the cyclic rank-pattern labels, the \(K_S\)-orbit
stratification, the Zariski closure relation, or the codimension computations.
\end{remark}
\section{Discussion}
Have shown that the critical points can be stratified into strata parameterized by geometric invariant: cyclic rank patterns. This stratification provides a combinatorial description of the loss landscape of DLNs. This is useful as a coarse grained description of the loss landscape. The loss landscape can be thought as a graph that is hierachically structured with particular nodes corresponding to a strata given by the rank $r$, the subset $S$ of singular vectors and cyclic rank patterns $\rho\in \mathcal{R}^S_d$. The stratification also induce some constraint on the training dynamics. The strata with high rank patterns are less degenerate and over a given $S$-critical loci the learning process can only move toward more degenerate strata which generalise the results from \cite{pepin}. We also discover geometrically principled observables: rank patterns. Rank patterns are useful observables to discuss the implicit biases of the learning process, for example SGD. Indeed using the rank patterns we can make explicit an implicit bias of SGD during training which over a given critical loci will favour solution with certain rank patterns. For example in the case of DLN at small initialization we expect from \cite{saxe} that SGD will have an implicit biases toward more degenerate strata i.e. solutions with low rank patterns. We should be able to make these implicit biases explicit by showing that SGD reduce some free energy functional of the student $W$ with an entropic regularization depending on the cyclic rank patterns or some corresponding continuous version using the singular values of partial product of weight matrices of the DLN.

\subsection{Detailed proof of Proposition~\ref{prop:S-critical-fiber}}
\label{app:proof-S-critical-fiber}

We prove the equality
\[
\mathcal C_d^S
=
\Gamma_d^S\cap\mu^{-1}(P_SW^\ast)
=
\iota_S^{-1}(\Gamma_d^r)\cap\mu^{-1}(P_SW^\ast)
\]
as an equality of subsets of \(\Omega\).

Recall that
\[
\iota_S(\theta)=(\theta,\Sigma_{XY}P_Q),
\]
and therefore
\[
\Gamma_d^S=\iota_S^{-1}(\Gamma_d^r).
\]
Thus \(\theta\in\Gamma_d^S\) means that
\[
(\theta,\Sigma_{XY}P_Q)\in\Gamma_d^r.
\]
Equivalently,
\[
\operatorname{rank}(\mu(\theta))=r
\]
and
\[
W_{<\ell}\Sigma_{XY}P_QW_{>\ell}=0
\qquad
\text{for all }1\leq \ell\leq L.
\]

We first prove
\[
\mathcal C_d^S
\subseteq
\Gamma_d^S\cap\mu^{-1}(P_SW^\ast).
\]
Let \(\theta\in\mathcal C_d^S\). By definition of the \(S\)-critical locus,
\[
\nabla_\theta\mathcal L(\theta)=0
\qquad
\text{and}
\qquad
\mu(\theta)=P_SW^\ast.
\]
Since \(\operatorname{rank}(P_SW^\ast)=r\), it follows that
\[
\operatorname{rank}(\mu(\theta))=r.
\]
It remains to show the cyclic equations with closing arrow \(\Sigma_{XY}P_Q\).

For each layer \(\ell\), the transpose-gradient formula is
\[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
2W_{<\ell}
\bigl(\Sigma_X\mu(\theta)^\top-\Sigma_{XY}\bigr)
W_{>\ell}.
\]
Using \(\mu(\theta)=P_SW^\ast\), we get
\[
\Sigma_X\mu(\theta)^\top-\Sigma_{XY}
=
\Sigma_X(P_SW^\ast)^\top-\Sigma_{XY}.
\]
Since \(P_S\) is an orthogonal projector, \(P_S^\top=P_S\), and therefore
\[
(P_SW^\ast)^\top=(W^\ast)^\top P_S.
\]
Moreover, by the definition
\[
W^\ast=\Sigma_{YX}\Sigma_X^{-1},
\]
and by symmetry of \(\Sigma_X\), we have
\[
\Sigma_X(W^\ast)^\top
=
\Sigma_X(\Sigma_X^{-1})^\top\Sigma_{YX}^\top
=
\Sigma_X\Sigma_X^{-1}\Sigma_{XY}
=
\Sigma_{XY}.
\]
Hence
\[
\Sigma_X(P_SW^\ast)^\top-\Sigma_{XY}
=
\Sigma_X(W^\ast)^\top P_S-\Sigma_{XY}
=
\Sigma_{XY}P_S-\Sigma_{XY}.
\]
Since \(P_Q=I-P_S\), this becomes
\[
\Sigma_X(P_SW^\ast)^\top-\Sigma_{XY}
=
-\Sigma_{XY}P_Q.
\]
Substituting into the gradient formula gives
\[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
-2W_{<\ell}\Sigma_{XY}P_QW_{>\ell}.
\]
Because \(\theta\in\mathcal C_d^S\), every partial gradient vanishes. Therefore
\[
W_{<\ell}\Sigma_{XY}P_QW_{>\ell}=0
\qquad
\text{for all }1\leq\ell\leq L.
\]
Thus
\[
(\theta,\Sigma_{XY}P_Q)\in\Gamma_d^r,
\]
or equivalently,
\[
\theta\in\Gamma_d^S.
\]
Since also \(\mu(\theta)=P_SW^\ast\), we have
\[
\theta\in\Gamma_d^S\cap\mu^{-1}(P_SW^\ast).
\]

Conversely, suppose that
\[
\theta\in\Gamma_d^S\cap\mu^{-1}(P_SW^\ast).
\]
Then
\[
\mu(\theta)=P_SW^\ast
\]
and, because \(\theta\in\Gamma_d^S\),
\[
W_{<\ell}\Sigma_{XY}P_QW_{>\ell}=0
\qquad
\text{for all }1\leq\ell\leq L.
\]
As above,
\[
\Sigma_X\mu(\theta)^\top-\Sigma_{XY}
=
-\Sigma_{XY}P_Q.
\]
Therefore, for every \(\ell\),
\[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
-2W_{<\ell}\Sigma_{XY}P_QW_{>\ell}
=
0.
\]
Hence
\[
\nabla_{W_\ell}\mathcal L(\theta)=0
\qquad
\text{for every }1\leq\ell\leq L.
\]
So
\[
\nabla_\theta\mathcal L(\theta)=0.
\]
Together with \(\mu(\theta)=P_SW^\ast\), this gives
\[
\theta\in\mathcal C_d^S.
\]
This proves the reverse inclusion and therefore the proposition.

\subsection{Detailed proof of Proposition~\ref{prop:critical-rank-pattern-decomposition}}
\label{app:proof-critical-rank-pattern-decomposition}

Assume that
\[
\Gamma_d^r=\bigsqcup_{\rho\in R}O_\rho^r
\]
is a finite decomposition into locally closed subsets of \(\Gamma_d^r\), and that
the pieces satisfy
\[
\overline{O_\rho^r}^{\,\Gamma_d^r}
\subseteq
\bigcup_{\rho'\leq\rho}O_{\rho'}^r.
\]
For each \(\rho\in R\), define
\[
\mathcal C_\rho^S
:=
\iota_S^{-1}(O_\rho^r)\cap\mu^{-1}(P_SW^\ast).
\]

We first prove that the pieces \(\mathcal C_\rho^S\) give a disjoint
decomposition of \(\mathcal C_d^S\). By Proposition~\ref{prop:S-critical-fiber},
\[
\mathcal C_d^S
=
\iota_S^{-1}(\Gamma_d^r)\cap\mu^{-1}(P_SW^\ast).
\]
Since
\[
\Gamma_d^r=\bigsqcup_{\rho\in R}O_\rho^r,
\]
we have
\[
\iota_S^{-1}(\Gamma_d^r)
=
\iota_S^{-1}\left(\bigsqcup_{\rho\in R}O_\rho^r\right).
\]
Preimages commute with arbitrary unions, and preimages of disjoint sets are
disjoint. Hence
\[
\iota_S^{-1}(\Gamma_d^r)
=
\bigsqcup_{\rho\in R}\iota_S^{-1}(O_\rho^r).
\]
Intersecting both sides with \(\mu^{-1}(P_SW^\ast)\) gives
\[
\mathcal C_d^S
=
\left(
\bigsqcup_{\rho\in R}\iota_S^{-1}(O_\rho^r)
\right)
\cap
\mu^{-1}(P_SW^\ast).
\]
Therefore
\[
\mathcal C_d^S
=
\bigsqcup_{\rho\in R}
\left(
\iota_S^{-1}(O_\rho^r)\cap\mu^{-1}(P_SW^\ast)
\right)
=
\bigsqcup_{\rho\in R}\mathcal C_\rho^S.
\]
After discarding the empty terms, this is the claimed decomposition.

Next we prove local closedness. Each \(O_\rho^r\) is locally closed in
\(\Gamma_d^r\). Since \(\Gamma_d^r\) is a locally closed subset of
\(\Omega\times \operatorname{Mat}_{d_0\times d_L}\), each \(O_\rho^r\) is locally
closed in \(\Omega\times \operatorname{Mat}_{d_0\times d_L}\). The map
\[
\iota_S:\Omega\longrightarrow \Omega\times \operatorname{Mat}_{d_0\times d_L}
\]
is continuous, so
\[
\iota_S^{-1}(O_\rho^r)
\]
is locally closed in \(\Omega\). The map \(\mu:\Omega\to
\operatorname{Mat}_{d_L\times d_0}\) is regular, hence continuous, and the point
\(\{P_SW^\ast\}\) is closed. Therefore
\[
\mu^{-1}(P_SW^\ast)
\]
is closed in \(\Omega\). Consequently
\[
\mathcal C_\rho^S
=
\iota_S^{-1}(O_\rho^r)\cap\mu^{-1}(P_SW^\ast)
\]
is locally closed in \(\Omega\). Since
\[
\mathcal C_\rho^S\subseteq \mathcal C_d^S,
\]
it is also locally closed in \(\mathcal C_d^S\) with the subspace topology.

It remains to prove the frontier inclusion
\[
\overline{\mathcal C_\rho^S}^{\,\mathcal C_d^S}
\subseteq
\bigcup_{\rho'\leq\rho}\mathcal C_{\rho'}^S.
\]
We use the following elementary fact. If \(f:X\to Y\) is continuous and
\(A\subseteq Y\), then
\[
\overline{f^{-1}(A)}^{\,X}
\subseteq
f^{-1}\left(\overline A^{\,Y}\right).
\]
Indeed, \(f^{-1}(\overline A^{\,Y})\) is closed in \(X\) and contains
\(f^{-1}(A)\), so it contains the closure of \(f^{-1}(A)\).

Apply this fact to the continuous map
\[
\iota_S|_{\mathcal C_d^S}:\mathcal C_d^S\longrightarrow \Gamma_d^r.
\]
This restriction is well-defined because, by Proposition~\ref{prop:S-critical-fiber},
\[
\mathcal C_d^S\subseteq\Gamma_d^S=\iota_S^{-1}(\Gamma_d^r),
\]
so \(\iota_S(\theta)\in\Gamma_d^r\) for every \(\theta\in\mathcal C_d^S\).

Now observe that
\[
\mathcal C_\rho^S
=
(\iota_S|_{\mathcal C_d^S})^{-1}(O_\rho^r).
\]
Therefore
\[
\overline{\mathcal C_\rho^S}^{\,\mathcal C_d^S}
=
\overline{(\iota_S|_{\mathcal C_d^S})^{-1}(O_\rho^r)}^{\,\mathcal C_d^S}
\subseteq
(\iota_S|_{\mathcal C_d^S})^{-1}
\left(
\overline{O_\rho^r}^{\,\Gamma_d^r}
\right).
\]
Using the assumed frontier inclusion,
\[
\overline{O_\rho^r}^{\,\Gamma_d^r}
\subseteq
\bigcup_{\rho'\leq\rho}O_{\rho'}^r,
\]
we obtain
\[
\overline{\mathcal C_\rho^S}^{\,\mathcal C_d^S}
\subseteq
(\iota_S|_{\mathcal C_d^S})^{-1}
\left(
\bigcup_{\rho'\leq\rho}O_{\rho'}^r
\right).
\]
Preimages commute with unions, so
\[
(\iota_S|_{\mathcal C_d^S})^{-1}
\left(
\bigcup_{\rho'\leq\rho}O_{\rho'}^r
\right)
=
\bigcup_{\rho'\leq\rho}
(\iota_S|_{\mathcal C_d^S})^{-1}(O_{\rho'}^r).
\]
But
\[
(\iota_S|_{\mathcal C_d^S})^{-1}(O_{\rho'}^r)
=
\mathcal C_{\rho'}^S.
\]
Hence
\[
\overline{\mathcal C_\rho^S}^{\,\mathcal C_d^S}
\subseteq
\bigcup_{\rho'\leq\rho}\mathcal C_{\rho'}^S.
\]
This proves the proposition.

The argument gives an inclusion rather than an equality. The possible failure of
equality comes from the strictness of
\[
\overline{f^{-1}(A)}^{\,X}
\subseteq
f^{-1}\left(\overline A^{\,Y}\right)
\]
for a general continuous map \(f\). In the present setting, a point in
\(\iota_S^{-1}(\overline{O_\rho^r}^{\,\Gamma_d^r})\) may be reachable as a
degeneration of \(O_\rho^r\) in the full cyclic variety while not being reachable
by a degeneration that remains inside the fixed specialization determined by
\(\iota_S\). Therefore the natural statement for the \(S\)-critical locus is the
one-sided frontier condition above.
\subsection{Proof of the Inversion Formula}
\label{app:inversion}

We give a slightly more explicit version of the proof of
Proposition~\ref{prop:inversion}. For every admissible interval \((a,b)\), the
forward formula gives
\[
\rho_{ab}
=
\sum_{u\leq a\leq b\leq v} m_{uv}.
\]
Therefore
\[
\rho_{ab}-\rho_{a-1,b}-\rho_{a,b+1}+\rho_{a-1,b+1}
=
\sum_{u,v} m_{uv}\,\Xi_{uv},
\]
where
\[
\Xi_{uv}
:=
\mathbf 1_{u\leq a\leq b\leq v}
-\mathbf 1_{u\leq a-1\leq b\leq v}
-\mathbf 1_{u\leq a\leq b+1\leq v}
+\mathbf 1_{u\leq a-1\leq b+1\leq v}.
\]

If \(u<a\), then the first two indicators cancel, and likewise the last two
indicators cancel. If \(v>b\), then the first and third indicators cancel, and
likewise the second and fourth indicators cancel. The only way for a nonzero
coefficient to remain is therefore
\[
u=a,\qquad v=b.
\]
In that case
\[
\Xi_{ab}=1.
\]
Thus the entire sum reduces to \(m_{ab}\), proving the inversion formula.

1,& \text{if there exists }t\in\mathbb Z
\text{ such that }
j+tn\le i\le j+b-1+tn\le i+a-1,\\
0,& \text{otherwise.}
\end{cases}
\]
Equivalently,
\[
\epsilon(I,J)=\dim \Hom_A(M_I,M_J).
\]
For \(m\in\mathcal M_d^{r,q}\), put
\[
\mathfrak e(m)
:=
\sum_{I,J\in\mathcal I_L}
m_I m_J\,\epsilon(I,J).
\]
Thus \(\mathfrak e(m)=\dim \End_A(M_m)\), where
\[
M_m:=\bigoplus_{I\in\mathcal I_L}M_I^{\oplus m_I}.
\]

\begin{theorem}[Codimension of the fixed-\(S\) critical locus]
\label{thm:fixed-S-critical-codim}
Assume that \(\mathcal C_d^S\neq \varnothing\). Then
\[
\boxed{
\codim_{\Omega}\mathcal C_d^S
=
D_d+p_{r,q}-g_d
+
\min_{m\in\mathcal M_d^{r,q}}\mathfrak e(m)
}
\]
where
\[
p_{r,q}
:=
r(d_0+d_L-r)
+
q(d_0+d_L-2r-q).
\]
Equivalently, in the full-output-rank case \(q=d_L-r\), this simplifies to
\[
\boxed{
\codim_{\Omega}\mathcal C_d^S
=
D_d+d_0d_L-g_d
+
\min_{m\in\mathcal M_d^{r,d_L-r}}\mathfrak e(m).
}
\]
Moreover, over \(\mathbb C\), the number of top-dimensional irreducible
components of \(\mathcal C_d^S\) is
\[
\#\left\{
m\in\mathcal M_d^{r,q}
\ \middle|\
\mathfrak e(m)
=
\min_{m'\in\mathcal M_d^{r,q}}\mathfrak e(m')
\right\}.
\]
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:S-critical-fiber},
\[
\mathcal C_d^S
=
\left\{
\theta\in\Omega
\ \middle|\
(\theta,E_S)\in\Gamma_d,\ \mu(\theta)=B_S
\right\}.
\]
Equivalently, if
\[
\pi:\Gamma_d\longrightarrow
\Hom(k^{d_0},k^{d_L})\times \Hom(k^{d_L},k^{d_0}),
\qquad
\pi(\theta,W_0):=(\mu(\theta),W_0),
\]
then
\[
\mathcal C_d^S=\pi^{-1}(B_S,E_S).
\]

Let
\[
\mathcal P_{r,q}
:=
\left\{
(B,E)
\ \middle|\
\rank B=r,\ \rank E=q,\ EB=0,\ BE=0
\right\}.
\]
The group
\[
G_{\mathrm{out}}:=\GL_{d_0}\times \GL_{d_L}
\]
acts transitively on \(\mathcal P_{r,q}\). Indeed, such a pair is equivalent to
the data of a rank-\(r\) map
\[
B:k^{d_0}\to k^{d_L}
\]
and a rank-\(q\) map
\[
E:k^{d_L}/\im B\longrightarrow \ker B.
\]
Hence
\[
\dim \mathcal P_{r,q}
=
r(d_0+d_L-r)
+
q\bigl((d_L-r)+(d_0-r)-q\bigr),
\]
which is the stated number \(p_{r,q}\).

The cyclic variety has the orbit decomposition
\[
\Gamma_d=\bigsqcup_{m\in\mathcal M_d^+}O_m,
\]
where \(O_m\) is the \(G_d\)-orbit of the module \(M_m\). The stabilizer of
\(M_m\) is \(\Aut_A(M_m)\), which is Zariski open in \(\End_A(M_m)\). Therefore
\[
\dim O_m
=
\dim G_d-\dim\End_A(M_m)
=
g_d-\mathfrak e(m).
\]

Now fix \(m\). The image \(\pi(O_m)\) is a single \(G_{\mathrm{out}}\)-orbit,
because \(\pi\) is \(G_d\)-equivariant and the projection
\(G_d\to G_{\mathrm{out}}\) is surjective. The orbit \(O_m\) meets
\(\pi^{-1}(B_S,E_S)\) if and only if
\[
m_{(0,L+1)}=r
\qquad\text{and}\qquad
\sum_{I\in\mathcal I_L}m_I\chi_{\mathrm{cl}}(I)=q,
\]
that is, if and only if \(m\in\mathcal M_d^{r,q}\). For such \(m\), the image is
\(\mathcal P_{r,q}\). Hence
\[
\dim\bigl(O_m\cap \pi^{-1}(B_S,E_S)\bigr)
=
\dim O_m-\dim\mathcal P_{r,q}
=
g_d-\mathfrak e(m)-p_{r,q}.
\]

Since
\[
\mathcal C_d^S
=
\bigsqcup_{m\in\mathcal M_d^{r,q}}
\left(O_m\cap \pi^{-1}(B_S,E_S)\right),
\]
we get
\[
\dim \mathcal C_d^S
=
\max_{m\in\mathcal M_d^{r,q}}
\left(g_d-\mathfrak e(m)-p_{r,q}\right)
=
g_d-p_{r,q}
-
\min_{m\in\mathcal M_d^{r,q}}\mathfrak e(m).
\]
Since \(\dim\Omega=D_d\), this gives
\[
\codim_{\Omega}\mathcal C_d^S
=
D_d-\dim\mathcal C_d^S
=
D_d+p_{r,q}-g_d
+
\min_{m\in\mathcal M_d^{r,q}}\mathfrak e(m).
\]

Finally, each nonempty piece
\[
O_m\cap \pi^{-1}(B_S,E_S)
\]
is an orbit of the connected algebraic group
\[
\{g\in G_d\mid (g_0,g_L)\cdot(B_S,E_S)=(B_S,E_S)\}.
\]
Hence each such piece is irreducible. Its closure gives a top-dimensional
irreducible component precisely when its dimension is maximal, equivalently
when \(\mathfrak e(m)\) is minimal. This proves the component count.
\end{proof}

\begin{remark}[Rank-pattern form]
Using the multiplicity--rank inversion formula, the same codimension formula can
be written as
\[
\codim_{\Omega}\mathcal C_d^S
=
D_d+p_{r,q}-g_d
+
\min_{\rho\in\mathcal R_d^{r,q}}
\sum_{I,J\in\mathcal I_L}
m_I(\rho)m_J(\rho)\epsilon(I,J),
\]
where
\[
\mathcal R_d^{r,q}
:=
\{\rho\in\mathcal R_d\mid \rho_{0L}=r,\ \rho_{L0}=q\}.
\]
Thus the formula is intrinsic to the cyclic rank-pattern stratification.
\end{remark}

\begin{corollary}[Recovery of the multiplication-fiber formula]
\label{cor:recover-simon}
If \(Q=\varnothing\), equivalently \(E_S=0\) and \(q=0\), then
\[
\mathcal C_d^S=\mu^{-1}(B_S).
\]
In this case Theorem~\ref{thm:fixed-S-critical-codim} becomes
\[
\codim_{\Omega}\mu^{-1}(B_S)
=
D_d+r(d_0+d_L-r)-g_d
+
\min_{m\in\mathcal M_d^{r,0}}\mathfrak e(m),
\]
which is the homogeneous-orbit form of the Simon--Rimanyi codimension formula
for the fiber of the multiplication map over a rank-\(r\) matrix.

In particular, if the teacher has full output rank and \(|S|=d_L\), then
\(r=d_L\), \(q=0\), and
\[
\mathcal C_d^S=\mu^{-1}(W^\ast),
\qquad
\codim_{\Omega}\mathcal C_d^S=d_0d_L.
\]
Thus the fixed-\(S\) critical locus is exactly the global-minimum fiber, and the
codimension agrees with the full-rank specialization of the multiplication-fiber
result.
\end{corollary}
\section{Geometry: Rank pattern stratification}
 The $S$-critical loci is the intersection of the $S$-cyclic variety with the fiber of the parameter function map over the critical student $P_SW^\ast$.
\begin{proposition}
The $S$-critical loci satisfies:
\[
\mathcal C_{d}^S=
\left\{
\theta\in \Gamma_{d}^S 
\ \middle|\
\mu(\theta)=P_SW^\ast
\right\} = \Gamma_{d}^S \cap \mu^{-1}(P_SW^\ast) = \iota_S^{-1}(\Gamma_d^r) \cap \mu^{-1}(P_SW^\ast) 
\]
\end{proposition}
\begin{proof}
  We already know from proposition 1 in \cite{achour2024loss} that for a critical point $\theta$ associated with $S$ we have $\mu(\theta) = P_SW^{\star}$ which is rank $r$. In the other direction, assume that $\mu(\theta)=P_SW^{\star}$. Then recall the gradient equations 
  \[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
2\,W_{<\ell}\bigl(\Sigma_X\Sigma_X^{-1}\Sigma_{XY}P_S-\Sigma_{XY}\bigr)W_{>\ell} = 2\,W_{<\ell}\Sigma_{XY}P_QW_{>\ell} = 0
\]
In other words, $\theta$ is a critical point of rank $r$.  
\end{proof}
Now assume that we have a stratification of the cyclic variety $\Gamma^r_{d}$ i.e.:
\[
\Gamma^r_{d} = \bigsqcup_{\rho} O^r_{\rho}
\]   
Where $\rho$ runs over some finite subset of all the cyclic rank patterns. Furthermore, assume that we have the following closure condition:
\[
\bar{O}_{\rho} = \bigcup_{\rho'\leq \rho} O_{\rho'}  
\] 
Coming from the following frontier condition:
\[
\bar{O}_{\rho}\cap O_{\rho’} \neq \emptyset \implies  O_{\rho’} \subset \bar{O}_{\rho}
\]
where $\rho' \leq \rho$ means that $\rho'_{ij} \leq \rho_{ij}$ for all $i,j$. This stratification induces a stratification over $\Gamma^r_{d,S}$ and therefore on the $S$-critical loci $\mathcal C_{d}^S$. 
\begin{proposition}
  Consider the stratification of $\Gamma^r_{d}$ by the strata $O^r_{\rho}$ satisfying a closure condition. The critical loci admits the following stratification
  \[
  \mathcal{C}^{S}_{d} = \bigsqcup_{\rho} \mathcal{C}^S_\rho
  \]
  Where we define:
  \[
  \mathcal{C}^S_\rho :=  \iota_S^{-1}(O^r_{\rho}) \cap \mu^{-1}(P_SW^{\ast})   
  \]
  Furthermore, the strata $\mathcal{C}^S_\rho$ are locally closed and satisfy the following condition:
  \[
  \bar{\mathcal{C}}_{\rho} \subset \bigcup_{\rho'\leq \rho} \mathcal{C}_{\rho'}  
  \] 
\end{proposition}
In other words the critical loci admits a combinatorial description by admissible cyclic rank patterns. This combinatorial description goes beyond providing a labelling of the critical points since the inclusion condition above constrains the transitions that are allowed between rank patterns. 
\begin{proof}[Proof Sketch] The stratification is induced by the intersection of the strata $\iota^{-1}(O^r_{\rho})$ with the fibers of the product map $\mu$. Each strata is locally closed as an intersection of a locally close sets with a closed set (the fiber of the multiplication map and the fiber of the inclusion map over a locally closed strata). The closure inclusion is inherited from the closure condition of the $O^{r}_{\rho}$. Indeed we have:
\[
\overline{\iota_S(O^{r}_{\rho})} \subset \iota_S^{-1}(\overline{O^r_{\rho}}) = \iota_S^{-1}(\bigcup_{\rho'\leq \rho} O^r_{\rho'}) = \bigcup_{\rho'\leq \rho} \iota_S^{-1}(O^r_{\rho'})
\]
And the inclusion follows from intersecting the latter inclusion with the fiber of the multiplication map over $P_SW^{\star}$.

\end{proof}

\subsection{Quiver representation of the cyclic variety}
The cyclic variety is equivalent to a representation of a cyclic quiver with finite relation.
\begin{definition}
  Define the cyclic quiver $Q=C_{L+1}$ with vertices $l\in\{0,...,L\}$ and edges $a_l: l-1\longrightarrow l \ \text{mod} L + 1$ with $a_0:L \longrightarrow 0$. Define the paths of length $L$ for $l\in\{1,...,L\}$: $$p_l:= a_{l-1}...a_1a_0a_L...a_{l+1} \ \text{mod} L+1$$. 
\end{definition}
Define the bounded path algebra with relations by quotienting the path algebra $kQ$ by imposing that all paths of length $L$ going through the closing arrow $a_0$ vanish.
\begin{definition} Let $k$ be a field, consider the path algebra $kQ$, consider the ideal:
\[
I = \langle p_1,...,p_L \rangle
\]
Define the algebra $A$ as the quiver $kQ$ with finite relations:
\[
A = kQ/I
\]
\end{definition}
There is an equivalence of categories between $A$-Modules and representations of $C_{L+1}$ with vanishing linear maps:
\[
\phi_{l-1}...\phi_0...\phi_{l+1} = 0
\]
We note $\Rep_d(A)$ the representations of $C_{L+1}$ with vanishing relations generated by $I$ and identify it with the category of $A$-modules. One also observe that the space of $A$-modules is equivalent to the cyclic variety $\Gamma_d$. 
\begin{proposition}
There cyclic variety $\Gamma_d \cong \Rep_d(A)$ via the identification:
\[
W_l = \phi_l \qquad V_i \cong \mathbb{R}^{d_i}
\] 
\end{proposition}
An important property is that $A$ is a Nakayama algebra. In other words any projective $A$-module has a unique composition series of finite lenght. This will enables us to classify all indecomposable representations of $A$ and write a Krull-Schmidt decomposition of any $A$-module M in terms of its indecomposable. This property is important as will enable us to later stratify the cyclic variety $\Gamma_d$ using the indecomposable representations of $A$.


Projective modules generalise the notion of a basis of a vector space for $kQ$ modules where $Q$ is a quiver. For example the module $P_i = Ae_i$ of paths starting at vertex $e_i$ is projective. The Jacobson radical will help us showing that any projective module has a unique composition series.
\begin{definition}
  A module $P$ is projective if for all surjective map $N\longrightarrow M$ the map $P\longrightarrow M$ lifts into a map $P \longrightarrow N$. 
\end{definition}
\begin{definition}
  Let $A$ be a $k$-algebra. The Jacobson radical $J:=J(A)$ is the intersection of all maximal ideals of $A$. 
\end{definition}
For example when $A=kQ/I$ then $J(A)$ is the set of path of length greater than 1 and for a projective module $P_i$ the quotient $P_i/A$ is simply $e_i$. The quotient space $A/J(A)$ is the algebra of vertices (paths of length 0). 
\begin{definition}
  An algebra $A$ is a Nakayama algebra if any projective $A$-module $P$ admits a unique decomposition series of finite length $m$:
  \[
  0 \subset P_{m} \subset \dots \subset P_1 \subset P
  \]
with $P_i/P_{i+1}$ is simple\footnote{A representation is simple if it has no non trivial subrepresentations}. The length $m=l(P)$ is called the Loewy length which is the smallest integer $m$ such that $J^m=0$. Let $(e_i)_{i\in Q_0}$ be the primitive indepempotent of $A$. The Kupisch series of $A$ is the tuple:
\[
(c_0,\ldots,c_{L}),\qquad c_i:=\ell(P_i),
\]
\end{definition}
\begin{proposition}
The algebra $A$ is finitely generated. Furthermore any indecomposable projective $P_i$ admits a unique composition series of finite length.
\end{proposition}
\begin{proof}
  The algebra $A$ is finitely generated by construction since all paths of length strictly greater than $L+1$ vanish. Thus any $A$-module $M$ is also finitely generated. Consider the projective module $P_i := Ae_i$. Consider the Jacobson radical $J$. Construct the submodule $P^l_i=J^lP_i$ of paths starting from $e_i$ with length greater or equal to $l$. Then the projective module $P_i$ admits the following composition series for $i\neq 0$:
  \[
  P_i \supseteq P_i^1 \supseteq ... \supseteq P_i^{L+1} = P^{L}_i = 0
  \]
Since all paths of length at least $L$ vanish. For $i=0$ all paths of length at least $L+1$ vanish so we have:
\[
  P_i \supseteq P_i^1 \supseteq ... \supseteq P_i^{L+2} = P^{L+1}_i = 0
\]
Note that the space $P_i/P_i^{l+1}$ correspond to the of length strictly less than $l+1$ starting at $e_i$. Hence, the quotient space $P_i^l/P^{l+1}_i$ correspond to all the vertices between $e_i$ and $e_l$ i.e. these modules are simple. Furthermore since $A$ is finitely generated then we have that all projective indecomposables are given by the $P_i$.

It remains to be shown that for each $P_i$, the composition series above is unique.

Since all path vanishes after some length $L$ or $L+1$, and there is no loop, nor branching in the quiver $C_{L+1}$, all the projective submodules of $P_i$ are of the form
\[
P_i^l
\]
For some $l< L+1$. Furthermore, the composition series of $P_i$ already contains all the submodules of $P_i$ and therefore it is unique for each $P_i$. 
\end{proof}
As a consequence of the finite unique composition series of all projective indecomposables we find that $A$ is a Nakayama algebra:
\begin{proposition}
  The algebra $A=kQ/I$ is a Nakayama algebra. Since the quiver $Q=C_{L+1}$ is cyclic with no branching and $A$ is finitely generated, we say that $A$ is a cyclic finite Nakayama algebra. 
\end{proposition}
\begin{definition}
  The interval modules starting at $i$ and ending at $j$ of length $\delta(i,j) = j-i \ \text{mod} \ L + 1$ are given by:
\[
M(i,j)_v := 
\begin{cases}
  k & \quad \text{if vertex} \ v\in [i\to j] \\
  0 & \quad \text{otherwise}
\end{cases}
\] 
And the linear maps are given by:
\[
M(i,j)_e := 
\begin{cases}
  id_k & \quad \text{if edge} \ e\subset [i\to j] \\
  0 & \quad \text{otherwise}
\end{cases}
\] 
\end{definition}
We have the following classification theorem for the indecomposables representation of the cyclic quiver $C_{L+1}$:
\begin{theorem}
  Let $M$ be an indecomposable $A$-modules then there is an interval module $M(i,j)$ such that: 
  \[
  M \cong M(i,j)
  \]
\end{theorem}
\begin{proof}
Let \(J=J(A)\). Since \(A\) is a finite-dimensional Nakayama algebra, a
standard theorem for Nakayama algebras implies that every
indecomposable \(A\)-module is uniserial (there is no branching in the quiver).

For each vertex \(i\), let \(P_i=Ae_i\) be the corresponding indecomposable
projective. Since \(P_i\) is uniserial, its submodules are precisely
\[
P_i,\ JP_i,\ J^2P_i,\ldots,\ J^{c_i}P_i=0,
\]
where \(c_i=\ell(P_i)\). Hence every quotient of \(P_i\) is of the form
\[
P_i/J^\ell P_i,\qquad 0\leq \ell\leq c_i.
\]

Conversely, let \(M\) be an indecomposable \(A\)-module. Since \(M\) is
uniserial then $\operatorname{rad} M=JM$ is maximal and hence its top is simple:
\[
\operatorname{top}M:=M/JM\cong S_i=P_i/JP_i
\]
for some vertex \(i\). Then $\text{top} \ P\cong \text{top} \ M \cong S_i$ which forces the projective cover of \(M\) to be
\[
P_i\twoheadrightarrow M.
\]
Writing \(K\) for the kernel, we have \(M\cong P_i/K\). Since \(K\subseteq P_i\)
and \(P_i\) is uniserial, \(K=J^\ell P_i\) for some \(\ell\). Therefore
\[
M\cong P_i/J^\ell P_i.
\]
These modules are exactly the interval modules
\(M(i,j)\) with \(j=i+\ell-1\).
\end{proof}
