\section{Setup and formal statement}
The critical loci of rank $r$ critical points in the loss landscape of a deep linear network admits a stratification parametrized by the rank of the critical points, rank patterns and a subset $S$ of singular vectors of the teacher matrix $M$.
\begin{definition}[Determinantal, cyclic, and critical loci]
Define
\[
\Sigma_d^r:=\{\theta\in \Omega:\rk(\mu(\theta))=r\} \quad \text{\textit{Determinantal variety}},
\] 
\[
\Gamma_{d}
:=
\left\{
(\theta,W_0)\in \Omega \times \text{Mat}_{d_0 \times d_L}
\ \middle|\
W_{<\ell}W_{0}W_{>\ell}=0
\text{ for all }1\leq \ell\leq L
\right\}, \quad \text{\textit{Cyclic variety}}
\]
Let $\pi:\Omega \times \text{Mat}_{d_0 \times d_L} \longrightarrow \text{Mat}_{d_0 \times d_L}$ be the natural projection. Define:
\[
\Gamma_{d}^r:=\Gamma_{d}\cap \Sigma_d^r, \qquad \Gamma_{d,S}^{r}:=\Gamma_{d}^r\cap\pi^{-1}(\Sigma_{XY}P_Q) = \left\{
\theta\in \Omega
\ \middle|\
W_{<\ell}\Sigma_{XY}P_QW_{>\ell}=0
\text{ for all }1\leq \ell\leq L
\right\}
\]
Let $S$ be a subset of left-singular vectors of $M$, define the $S$-critical loci:
\[
\mathcal C^S_{d} = \{\theta\in\Omega | \ \nabla_{\theta} L(\theta)=0; \mu(\theta)= P_SW^{\star}\}
\]
\end{definition}
\begin{definition}\label{def:cyclic-rank-pattern}
For $\phi = (\theta,W_0)\in\Gamma_{d}$ we define its \emph{cyclic rank pattern} to be the array
\[
r(\phi)=(r_{ij}(\phi))_{0\le i,j\le L}
\quad\text{where}\quad
r_{ij}(\phi):=\mathrm{rank}\bigl(W_{j}...W_i\bigr) \quad \text{for} \quad \text{and} \quad r_{ii}(\phi) = d_i \text{and} \quad r_{i+1i}(\phi) = \mathrm{rank}\bigl(W_i\bigr)
\]
\end{definition}
\begin{theorem}
Define the strata $\mathcal O^S_{\rho} := \mathcal O_{\rho} \cap \Gamma^r_{d,S}$ where  $\mathcal{O}_{\rho}$ is a $G_d$-orbit of the cyclic variety $\Gamma^r_{d}$.
Assuming that the data is non degenerate and the input covariance matrix is invertible then the loci of critical points of rank $r<r_\text{max}$ admits the following stratification parameterized by subset $S$ of singular vectors of the teacher with $r$ elements and admissible rank patterns $\rho\in R_d^{\mathrm{orb}}$:
\begin{align*}
  \mathcal{C}^r_d & = \bigsqcup_{|S|=r}\mathcal{C}^S_d = \bigsqcup_{|S|=r}\bigsqcup_{\substack{\rho\in R_d^{\mathrm{orb}}\\ \operatorname{rank}(\rho)=\operatorname{rank}(\Sigma_{XY}P_Q)}} \mathcal{O}^S_{\rho} \\
\end{align*}
\end{theorem}
\section{Geometry: Rank pattern stratification}
The $S$-critical loci is the intersection of the $S$-cyclic variety with the fiber of the parameter function map over the critical student $P_SW^\ast$.
\begin{proposition}
The $S$-critical loci satisfies:
\[
\mathcal C_{d}^S=
\left\{
\theta\in \Gamma_{d,S}^r
\ \middle|\
\mu(\theta)=P_SW^\ast
\right\} = \Gamma_{d,S}^r \cap \mu^{-1}(P_SW^\ast)
\]
\end{proposition}
\begin{proof}
  We already know from proposition 1 in \cite{achour2024loss} that for a critical point $\theta$ associated with $S$ we have $\mu(\theta) = P_SW^{\star}$. In the other direction, assume that $\mu(\theta)=P_SW^{\star}$. Then recall the gradient equations 
  \[
\bigl(\nabla_{W_\ell}\mathcal L(\theta)\bigr)^\top
=
2\,W_{<\ell}\bigl(\Sigma_X\Sigma_X^{-1}\Sigma_{XY}P_S-\Sigma_{XY}\bigr)W_{>\ell} = 2\,W_{<\ell}\Sigma_{XY}P_QW_{>\ell} = 0
\]
In other words, $\theta$ is a critical point. 
\end{proof}
Now assume that we have a stratification of the cyclic variety $\Gamma^r_{d}$ i.e.:
\[
\Gamma^r_{d} = \bigsqcup_{\rho} O^r_{\rho}
\]   
Where $\rho$ runs over some finite set of rank patterns. Furthermore, assume that we have the following closure condition:
\[
\bar{O}_{\rho} = \bigcup_{\rho'\leq \rho} O_{\rho'}  
\] 
Coming from the following frontier condition:
\[
\bar{O}_{\rho}\cap O_{\rho’} \neq \emptyset \implies  O_{\rho’} \subset \bar{O}_{\rho}
\]
where $\rho' \leq \rho$ means that $\rho'_{ij} \leq \rho_{ij}$ for all $i,j$. This stratification induces a stratification over $\Gamma^r_{d,S}$ and therefore on the $S$-critical loci $\mathcal C_{d}^S$. 
\begin{proposition}
  Consider the stratification of $\Gamma^r_{d}$ by the strata $O^r_{\rho}$ satisfying a closure condition. The critical loci admits the following stratification
  \[
  \mathcal{C}^{r}_{d,S} = \bigsqcup_{\rho} \mathcal{C}^S_\rho := \bigsqcup_{\rho}  O^r_{\rho} \cap \mu^{-1}(P_SW^{\ast}) \cap \pi^{-1}(\Sigma_{XY}P_Q)    
  \]
  Furthermore, the strata $\mathcal{C}^S_\rho$ are locally close and satisfy a closure condition:
  \[
  \bar{\mathcal{C}}_{\rho} = \bigcup_{\rho'\leq \rho} \mathcal{C}_{\rho'}  
  \] 
\end{proposition}
In other words the critical loci admits a combinatorial description by admissible cyclic rank patterns. This combinatorial description goes beyond providing a labelling of the critical points since the closure condition constrain the transitions that are allowed between rank patterns as lower rank patterns lie at the boundary of larger rank patterns. 
\begin{proof}[Proof Sketch] The stratification is induced by the intersection of the strata with the fibers of the product map $\mu$ and projection $\pi$. Each strata is locally closed as an interesection of locally close sets with closed sets. The closure condition is inherited from the closure condition of the $O^{r}_{\rho}$.
\end{proof}
