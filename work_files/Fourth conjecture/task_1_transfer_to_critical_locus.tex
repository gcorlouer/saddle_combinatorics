\section*{Task 1: Transfer to the critical locus}

\paragraph{Goal.}
Combine the rank-pattern orbit classification of the cyclic locus with the
first-conjecture orbit correspondence.

\paragraph{Ambient rank-pattern strata.}
For
\[
\rho\in \widetilde{\mathcal R}_{d,S}^{\,r},
\]
let
\[
\Gamma_{d,S}^r[\rho]
:=
\{W\in \Gamma_{d,S}^r\mid \rho(W)=\rho\}.
\]
By the third-conjecture theorem, each nonempty
\(\Gamma_{d,S}^r[\rho]\) is a single \(G_d^{W_0}\)-orbit.

\paragraph{Critical rank-pattern pieces.}
Define
\[
\mathcal C_{d,S}^r[\rho]
:=
\mathcal C_{d,S}^r\cap \Gamma_{d,S}^r[\rho].
\]
The orbit-bijection theorem from the first conjecture implies that every
\(G_d^{W_0}\)-orbit in \(\Gamma_{d,S}^r\) meets \(\mathcal C_{d,S}^r\) in
exactly one \(H_S\)-orbit. Therefore:
\begin{enumerate}
\item every nonempty \(\Gamma_{d,S}^r[\rho]\) contributes a nonempty
  \(\mathcal C_{d,S}^r[\rho]\);
\item each \(\mathcal C_{d,S}^r[\rho]\) is a single \(H_S\)-orbit.
\end{enumerate}

\paragraph{Orbit-set formulation.}
The inclusion
\[
\mathcal C_{d,S}^r\hookrightarrow \Gamma_{d,S}^r
\]
therefore induces a bijection
\[
H_S\backslash \mathcal C_{d,S}^r
\xrightarrow{\sim}
G_d^{W_0}\backslash \Gamma_{d,S}^r
\xrightarrow{\sim}
\widetilde{\mathcal R}_{d,S}^{\,r}.
\]

\paragraph{Status.}
Completed. Every admissible cyclic rank pattern now determines exactly one
critical \(H_S\)-orbit.
