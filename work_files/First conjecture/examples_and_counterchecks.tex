\section{Checks after Task 1}

\begin{example}[Low-depth check for the $G_d$-action, $L=2$]
For
\[
W=(W_1,W_2),
\qquad
g=(g_0,g_1,g_2)\in G_d,
\]
one has
\[
g\cdot W=(g_1W_1g_0^{-1},\, g_2W_2g_1^{-1}),
\]
so
\[
\mu(g\cdot W)
=(g_2W_2g_1^{-1})(g_1W_1g_0^{-1})
=g_2W_2W_1g_0^{-1}
=g_2\mu(W)g_0^{-1}.
\]
This is the smallest nontrivial instance of the product equivariance formula.
\end{example}

\begin{example}[Hidden gauge check, $L=3$]
Let
\[
h=(h_1,h_2)\in G_{\mathrm{hid}},
\]
so
\[
h\cdot W=(h_1W_1,\; h_2W_2h_1^{-1},\; W_3h_2^{-1}).
\]
Then
\[
\mu(h\cdot W)
=(W_3h_2^{-1})(h_2W_2h_1^{-1})(h_1W_1)
=W_3W_2W_1
=\mu(W).
\]
Thus the hidden-layer action only moves factorizations of a fixed end-to-end map.
\end{example}

\begin{example}[Cyclic equations under $G_d^{W_0}$, $L=2$]
For $L=2$ the cyclic equations are
\[
E_1(W)=W_0W_2,
\qquad
E_2(W)=W_1W_0.
\]
If $g\in G_d^{W_0}$, then
\[
E_1(g\cdot W)=g_0E_1(W)g_1^{-1},
\qquad
E_2(g\cdot W)=g_1E_2(W)g_2^{-1}.
\]
Hence the vanishing of the cyclic equations is preserved exactly as predicted by Task~1.
\end{example}

\section{Edge cases already settled}

\begin{remark}[The case $r=0$]
Task~0 showed that the critical locus is stable under the full group $G_d^{W_0}$ only in rank zero. This is an edge case for the orbit statement, not for the equivariance lemmas themselves.
\end{remark}

\begin{remark}[The case $r=d_L$]
If $r=d_L$, then $P_Q=0$, hence
\[
W_0=\Sigma_{XY}P_Q=0.
\]
Therefore the cyclic equations are automatic and
\[
\Gamma_{d,S}=\Omega.
\]
In this maximal-rank case, Task~1 reduces to ordinary stability of the determinantal locus under base change.
\end{remark}

\begin{remark}[Depth-one edge case]
If $L=1$, then the hidden gauge group is trivial and the single cyclic equation is
\[
E_1(W)=W_0.
\]
So the group-action formulas remain correct, but the cyclic locus is nonempty only when $W_0=0$. This is a degenerate depth regime rather than the main one of interest.
\end{remark}

\section{Where assumptions enter so far}

\begin{remark}
Up to and including Task~1:
\begin{itemize}
\item the distinct singular values assumption is not used;
\item the no-bottleneck assumption is not used;
\item the only inputs are the algebraic definitions of $\mu$, $W_0$, the cyclic equations, and the group actions.
\end{itemize}
\end{remark}

\section{Checks after Task 2}

\begin{example}[Endpoint lifting for $L=2$]
Let
\[
(g_0,g_2)\in K_S.
\]
Then
\[
\widetilde g:=(g_0,I_{d_1},g_2)\in G_d^{W_0}
\]
is a lift of the endpoint pair. Moreover,
\[
\mu(\widetilde g\cdot W)=g_2\mu(W)g_0^{-1}.
\]
So every endpoint normalization in $K_S$ is realized by an actual parameter-space move in $G_d^{W_0}$.
\end{example}

\begin{example}[Orbit image under $\mu$]
For any $W\in \Omega$, Task~2 proves
\[
\mu\bigl(G_d^{W_0}\cdot W\bigr)=K_S\cdot \mu(W).
\]
This means that the only information from a $G_d^{W_0}$-orbit relevant to Conjecture~A is the left-right orbit of its end-to-end map.
\end{example}

\begin{remark}[Rank-zero edge case]
When $r=0$, Task~0 already showed that the old quotient statement survives. Task~2 remains true in this case as well, but the endpoint reduction is less informative because the distinguished target product is just
\[
P_SW^\ast=0.
\]
\end{remark}

\begin{remark}[Assumptions still unused]
Task~2 is still purely representation-theoretic and uses no spectral genericity assumption and no no-bottleneck assumption. Those have not entered yet.
\end{remark}

\section{Checks after Task 3}

\begin{example}[Visible product-level constraints]
If $W\in \Gamma_{d,S}$ and $M=\mu(W)$, Task~3 proves
\[
W_0M=0
\qquad\text{and}\qquad
MW_0=0.
\]
Equivalently,
\[
\im(M)\subseteq \Ker(W_0)
\qquad\text{and}\qquad
\im(W_0)\subseteq \Ker(M).
\]
These are the constraints that genuinely descend to the end-to-end map.
\end{example}

\begin{example}[Positive-rank separation example]
Take
\[
L=3,
\qquad
\Sigma_{XY}=I_2,
\qquad
W_0=
\begin{pmatrix}
0&0\\
0&1
\end{pmatrix},
\qquad
M=
\begin{pmatrix}
1&0\\
0&0
\end{pmatrix}.
\]
Then
\[
\rk(M)=1,
\qquad
W_0M=0,
\qquad
MW_0=0.
\]
However, the two factorizations
\[
W=(M,M,I_2)
\qquad\text{and}\qquad
\widetilde W=(I_2,I_2,M)
\]
have the same product
\[
\mu(W)=\mu(\widetilde W)=M,
\]
but only the first one is cyclic. So the full cyclic locus cannot be reconstructed from $M$ alone.
\end{example}

\begin{remark}[Where assumptions enter after Task 3]
Task~3 uses the full-rank assumption on $\Sigma_{XY}$ only to sharpen
\[
\im(M)\subseteq \Ker(W_0)
\]
into
\[
\im(M)=\Ker(W_0)=\im(P_S)
\]
on the rank-$r$ sheet. The no-bottleneck assumption is still unused.
\end{remark}

\section{Checks after Task 4}

\begin{example}[A $2\times 2$ normal-form model]
Take
\[
\Sigma_{XY}=I_2,
\qquad
P_Q=
\begin{pmatrix}
0&0\\
0&1
\end{pmatrix},
\qquad
W_0=
\begin{pmatrix}
0&0\\
0&1
\end{pmatrix}.
\]
If
\[
M=
\begin{pmatrix}
a&b\\
c&d
\end{pmatrix}
\]
satisfies
\[
W_0M=0
\qquad\text{and}\qquad
MW_0=0,
\]
then necessarily
\[
c=d=0
\qquad\text{and}\qquad
b=d=0,
\]
so
\[
M=
\begin{pmatrix}
a&0\\
0&0
\end{pmatrix}.
\]
Thus every rank-one such matrix lies in the same endpoint orbit as
\[
\begin{pmatrix}
1&0\\
0&0
\end{pmatrix}=P_SW^\ast.
\]
This is the smallest concrete model of the Task~4 normalization theorem.
\end{example}

\begin{remark}[Where assumptions enter after Task 4]
Task~4 still does not use distinct singular values. It uses:
\begin{itemize}
\item the full-rank assumption on $\Sigma_{XY}$;
\item the existence of the model critical product $P_SW^\ast$;
\item the visible invariants extracted in Task~3.
\end{itemize}
\end{remark}

\section{Checks after Task 5}

\begin{example}[Task~5 in depth two]
Let
\[
L=2,
\qquad
W\in \Gamma_{d,S}^r.
\]
If Task~4 produces an endpoint pair
\[
(g_0,g_2)\in K_S
\]
with
\[
g_2\mu(W)g_0^{-1}=P_SW^\ast,
\]
then the lifted element
\[
g=(g_0,I_{d_1},g_2)\in G_d^{W_0}
\]
satisfies
\[
g\cdot W\in \Gamma_{d,S}^r
\qquad\text{and}\qquad
\mu(g\cdot W)=P_SW^\ast.
\]
This is the smallest concrete instance of the proof of Conjecture~A.
\end{example}

\begin{remark}[Rank-zero edge]
If $r=0$, then every point of $\Gamma_{d,S}^0$ has end-to-end map
\[
\mu(W)=0=P_SW^\ast.
\]
So Task~5 is trivial in this edge case: the identity element already gives a
critical representative.
\end{remark}

\begin{remark}[Where assumptions enter after Task 5]
Task~5 introduces no new assumptions beyond Tasks~1, 2, and~4. In particular,
the distinct singular values assumption and the no-bottleneck assumption remain
unused.
\end{remark}

\section{Checks after Task 7}

\begin{example}[Orbit intersection in the rank-zero edge case]
If
\[
r=0,
\]
then
\[
P_SW^\ast=0
\qquad\text{and}\qquad
H_S=G_d^{W_0}.
\]
So for every
\[
W\in \mathcal C_{d,S}^0,
\]
the equality
\[
\bigl(G_d^{W_0}\cdot W\bigr)\cap \mathcal C_{d,S}^0
=
H_S\cdot W
\]
is immediate. This is the simplest model of the Task~7 uniqueness statement.
\end{example}

\begin{remark}[What Task 7 adds]
Task~7 is purely an orbit-intersection argument. It uses equivariance and the
definition of the critical locus, but no new geometric or spectral input.
\end{remark}

\section{Checks after Task 8}

\begin{remark}[Task~8 is the assembly step]
Task~8 adds no new orbit computation beyond Tasks~5 and~7. Its content is that:
\begin{itemize}
\item Task~5 gives at least one critical representative in every ambient
$G_d^{W_0}$-orbit;
\item Task~7 shows that any two such representatives differ by $H_S$.
\end{itemize}
Therefore the ambient orbit set of $\Gamma_{d,S}^r$ is identified with the
critical orbit set of $\mathcal C_{d,S}^r$.
\end{remark}
